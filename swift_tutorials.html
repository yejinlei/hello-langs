1. 变量声明与初始化

Swift - 变量声明与初始化

特点：支持多种声明方式，包括变量(var)、常量(let)、类型标注和类型推导。注意事项：常量一旦赋值不可修改，Swift是类型安全语言，变量声明后类型不可更改。

# 变量声明与初始化
// 变量声明
var age Int
var name String
var isActive Bool
// 声明并初始化
var age = 25
var name = "张三"
var price Double = 19.99
// 常量声明
let pi = 3.14159
let maxUsers = 100
// 类型推导
var count = 10 // 自动推导为Int类型
var message = "Hello" // 自动推导为String类型

2. 基本数据类型

Swift - 基本数据类型

特点：包含丰富的内置类型，包括数值类型、布尔型、字符串型等。注意事项：不同类型的数值不能直接运算，需要显式转换，字符串是值类型。

# 基本数据类型
// 布尔型
var isActive Bool = true
var isAvailable = false
// 整型
var integer Int = 42
var int8 Int8 = 127
var uint UInt = 100
// 浮点型
var double Double = 3.14159
var float Float = 2.718
// 字符串
var str String = "Hello Swift"
var emptyString = ""
var multiLine = """多行 字符串"""
// 字符
var char Character = "A"

3. 控制流语句

Swift - 控制流语句

特点：包含条件语句、循环语句、控制转移语句等。注意事项：条件语句不需要括号，循环语句支持for-in、while、repeat-while等多种形式。

# 控制流语句
// if-else 语句
var score = 85
if score >= 90 {
    print("优秀")
} else if score >= 80 {
    print("良好")
} else {
    print("及格")
}
// switch 语句
var day = 3
switch day {
    case 1:
        print("星期一")
    case 2:
        print("星期二")
    default:
        print("其他天")
}
// for-in 循环
for i in 1...5 {
    print(i)
}
let numbers = [1, 2, 3, 4, 5]
for number in numbers {
    print(number)
}
// while 循环
var count = 0
while count < 5 {
    print(count)
    count += 1
}
// repeat-while 循环
var index = 0
repeat {
    print(index)
    index += 1
} while index < 5

4. 函数定义与使用

Swift - 函数定义与使用

特点：支持函数参数、返回值、默认参数、可变参数、in-out参数等。注意事项：函数参数默认是常量，in-out参数允许在函数内部修改外部变量。

# 函数定义与使用
// 基本函数
func greet(name: String) -> String {
    return "Hello, \(name)!"
}
// 调用函数
let message = greet(name: "张三")
// 无返回值函数
func printGreeting(name: String) {
    print("Hello, \(name)!")
}
// 默认参数
func greet(name: String, greeting: String = "Hello") -> String {
    return "\(greeting), \(name)!"
}
// 可变参数
func sum(numbers: Int...) -> Int {
    var total = 0
    for number in numbers {
        total += number
    }
    return total
}
// in-out 参数
func swap(_ a: inout Int, _ b: inout Int) {
    let temp = a
    a = b
    b = temp
}

5. 字符串与集合类型

Swift - 字符串与集合类型

特点：字符串支持丰富的操作，集合类型包括数组、集合和字典。注意事项：数组是有序集合，集合是无序唯一元素集合，字典是键值对集合。

# 字符串与集合类型
// 字符串操作
var str = "Hello Swift"
print(str.count) // 字符串长度
print(str.hasPrefix("Hello")) // 前缀检查
print(str.hasSuffix("Swift")) // 后缀检查
var combined = str + "!" // 字符串拼接
// 数组
var numbers = [1, 2, 3, 4, 5]
var strings: [String] = []
numbers.append(6) // 添加元素
numbers.remove(at: 0) // 移除元素
print(numbers.first) // 第一个元素
print(numbers.last) // 最后一个元素
// 集合
var set: Set<Int> = [1, 2, 3, 4, 5]
set.insert(6) // 添加元素
set.remove(1) // 移除元素
print(set.contains(2)) // 检查元素
// 字典
var dict: [String: Int] = [
    "张三": 25,
    "李四": 30
]
dict["王五"] = 35 // 添加键值对
dict.removeValue(forKey: "张三") // 移除键值对
print(dict["李四"]) // 获取值

6. 结构体和类

Swift - 结构体和类

特点：结构体是值类型，类是引用类型。注意事项：结构体默认有成员初始化器，类需要显式定义初始化器，引用类型使用引用计数进行内存管理。

# 结构体和类
// 结构体定义
struct Person {
    var name: String
    var age: Int
    let id: String
}
// 使用结构体
var person1 = Person(name: "张三", age: 25, id: "123456")
// 结构体是值类型
var person2 = person1
person2.name = "李四" // 修改person2不会影响person1
// 类定义
class Student {
    var name: String
    var grade: Int
    init(name: String, grade: Int) {
        self.name = name
        self.grade = grade
    }
    func study() {
        print("\(name) is studying.")
    }
}
// 使用类
let student1 = Student(name: "王五", grade: 100)
student1.study()
// 类是引用类型
let student2 = student1
student2.name = "赵六" // 修改student2会影响student1

7. 枚举

Swift - 枚举

特点：支持关联值、原始值、递归枚举等。注意事项：枚举可以有方法和属性，可以遵守协议，关联值可以携带额外信息。

# 枚举
// 基本枚举
enum Direction {
    case north
    case south
    case east
    case west
}
// 使用枚举
var direction = Direction.north
direction = .south
// 关联值枚举
enum Result {
    case success(Int)
    case failure(Error)
}
let result: Result = .success(42)
// 原始值枚举
enum Planet: Int {
    case mercury = 1
    case venus
    case earth
    case mars
}
let earth = Planet.earth.rawValue // 3
// 枚举方法
enum Weather {
    case sunny, rainy, cloudy
    func description() -> String {
        switch self {
        case .sunny:
            return "晴天"
        case .rainy:
            return "雨天"
        case .cloudy:
            return "多云"
        }
    }
}

8. 可选类型

Swift - 可选类型

特点：可选类型表示可能有值也可能没有值的类型。注意事项：可选类型需要解包才能使用，使用!强制解包可能导致运行时错误，推荐使用可选绑定或空合运算符。

# 可选类型
// 可选类型声明
var optionalString: String?
var optionalInt: Int? = 42
// nil 值
optionalString = nil
// 可选绑定
if let str = optionalString {
    print(str)
} else {
    print("字符串为空")
}
// 强制解包
let value = optionalInt! // 只有确定有值时才能使用
// 空合运算符
let result = optionalString ?? "默认值"
// 可选链
struct Person {
    var name: String?
    var age: Int?
}
var person: Person? = Person()
let personName = person?.name ?? "匿名"

9. 闭包

Swift - 闭包

特点：Swift中的闭包是自包含的功能代码块，可以在代码中传递和使用。注意事项：闭包会自动捕获其所在上下文中的变量和常量，避免强引用循环。

# 闭包的基本语法
// 完整语法的闭包
let add = { (a: Int, b: Int) -> Int in
    return a + b
}
// 调用闭包
let sum = add(5, 3) // 8
// 类型推断的闭包
let multiply: (Int, Int) -> Int = { a, b in
    return a * b
}
// 隐式返回的闭包
let subtract: (Int, Int) -> Int = { a, b in a - b }
// 缩写参数名的闭包
let divide: (Int, Int) -> Int = { $0 / $1 }
# 尾随闭包
// 接受闭包参数的函数
func performOperation(a: Int, b: Int, operation: (Int, Int) -> Int) -> Int {
    return operation(a, b)
}
// 正常调用
let result1 = performOperation(a: 10, b: 5, operation: { $0 - $1 })
// 尾随闭包语法
let result2 = performOperation(a: 10, b: 5) {
    return $0 * $1
}
// forEach方法使用尾随闭包
let numbers = [1, 2, 3, 4, 5]
numbers.forEach { number in
    print(number)
}
# 闭包捕获值
// 闭包捕获外部变量
var counter = 0
let increment = { in
    counter += 1
    print(counter)
}
increment() // 1
increment() // 2
increment() // 3
// 逃逸闭包
var completionHandlers: [() -> Void] = []
func someFunctionWithEscapingClosure(completionHandler: @escaping () -> Void) {
    completionHandlers.append(completionHandler)
}
func someFunctionWithNonescapingClosure(closure: () -> Void) {
    closure()
}
someFunctionWithEscapingClosure {
    print("逃逸闭包被调用")
}
completionHandlers.forEach { $0() } // 调用所有保存的闭包

10. 协议和扩展

Swift - 协议和扩展

特点：协议定义接口，扩展为现有类型添加功能。注意事项：协议可以被结构体、类、枚举遵守，扩展不能添加存储属性，只能添加计算属性和方法。

# 协议和扩展
// 协议定义
protocol Drawable {
    var color: String { get set }
    func draw()
}
// 结构体遵守协议
struct Circle: Drawable {
    var color: String
    var radius: Double
    func draw() {
        print("绘制一个\(color)色的圆，半径为\(radius)")
    }
}
// 扩展
extension Int {
    func isEven() -> Bool {
        return self % 2 == 0
    }
    var squared: Int {
        return self * self
    }
}
// 使用扩展
let number = 5
print(number.isEven()) // false
print(number.squared) // 25

11. 泛型

Swift - 泛型

特点：允许创建灵活的、可重用的函数和类型，能够处理任何类型的数据。注意事项：泛型函数和类型在定义时使用占位符类型，在使用时指定具体类型。

# 泛型
// 泛型函数
func swapTwoValues<T>(_ a: inout T, _ b: inout T) {
    let temporaryA = a
    a = b
    b = temporaryA
}
// 使用泛型函数
var someInt = 3
var anotherInt = 107
swapTwoValues(&someInt, &anotherInt) // someInt现在是107，anotherInt现在是3
var someString = "hello"
var anotherString = "world"
swapTwoValues(&someString, &anotherString) // someString现在是"world"，anotherString现在是"hello"
// 泛型类型
struct Stack<Element> {
    var items = [Element]()
    mutating func push(_ item: Element) {
        items.append(item)
    }
    mutating func pop() -> Element? {
        return items.popLast()
    }
}
// 使用泛型类型
var stackOfStrings = Stack<String>()
stackOfStrings.push("uno")
stackOfStrings.push("dos")
stackOfStrings.push("tres")

// 定义错误类型

13. 类型转换与类型检查

Swift - 类型转换与类型检查

特点：使用is关键字进行类型检查，使用as关键字进行类型转换。注意事项：类型转换可以是向上转型（子类型到父类型）或向下转型（父类型到子类型），向下转型可能失败。

# 类型转换与类型检查
// 类层次结构
class MediaItem {
    var name: String
    init(name: String) {
        self.name = name
    }
}
class Movie: MediaItem {
    var director: String
    init(name: String, director: String) {
        self.director = director
        super.init(name: name)
    }
}
class Song: MediaItem {
    var artist: String
    init(name: String, artist: String) {
        self.artist = artist
        super.init(name: name)
    }
}
let library = [
    Movie(name: "星球大战", director: "乔治·卢卡斯"),
    Song(name: "Hello", artist: "Adele"),
    Movie(name: "泰坦尼克号", director: "詹姆斯·卡梅隆"),
    Song(name: "Shape of You", artist: "Ed Sheeran")
]
// 类型检查
var movieCount = 0
var songCount = 0
for item in library {
    if item is Movie {
        movieCount += 1
    } else if item is Song {
        songCount += 1
    }
}
// 类型转换
for item in library {
    if let movie = item as? Movie {
        print("电影: \(movie.name)，导演: \(movie.director)")
    } else if let song = item as? Song {
        print("歌曲: \(song.name)，艺术家: \(song.artist)")
    }
}

14. 内存管理

Swift - 内存管理

特点：使用自动引用计数（ARC）管理内存，自动跟踪和管理对象的引用计数。注意事项：需要避免强引用循环，使用弱引用（weak）或无主引用（unowned）来解决。

# 内存管理
// 自动引用计数
class Person {
    var name: String
    init(name: String) {
        self.name = name
        print("\(name) 被初始化")
    }
    deinit {
        print("\(name) 被释放")
    }
}
var reference1: Person?
var reference2: Person?
var reference3: Person?
reference1 = Person(name: "张三")
reference2 = reference1
reference3 = reference1
reference1 = nil
reference2 = nil
reference3 = nil // 此时张三被释放
// 弱引用
class Apartment {
    var unit: String
    init(unit: String) {
        self.unit = unit
    }
    weak var tenant: Person?
    deinit {
        print("公寓 \(unit) 被释放")
    }
}
var john: Person?
var unit4A: Apartment?
john = Person(name: "John")
unit4A = Apartment(unit: "4A")
john?.apartment = unit4A
unit4A?.tenant = john
john = nil // John被释放
unit4A = nil // 公寓4A被释放

15. 访问控制

Swift - 访问控制

特点：提供不同级别的访问控制，用于限制代码的访问范围。注意事项：访问级别从高到低依次为public、internal、fileprivate、private。

# 访问控制
// public - 可以在任何地方访问
public class PublicClass {
    public var publicProperty: Int = 0
    internal var internalProperty: Int = 0
    fileprivate var fileprivateProperty: Int = 0
    private var privateProperty: Int = 0
    public func publicMethod() {
        // 实现
    }
    private func privateMethod() {
        // 实现
    }
}
// internal - 默认访问级别，只能在模块内部访问
class InternalClass {
    var property: Int = 0 // 默认为internal
}
// fileprivate - 只能在当前文件中访问
fileprivate class FileprivateClass {
    var property: Int = 0
}
// private - 只能在定义的声明中访问
class PrivateExample {
    private var privateProperty: Int = 0
    func accessPrivateProperty() {
        privateProperty += 1 // 可以在类内部访问
    }
}

16. 高级运算符

Swift - 高级运算符

特点：包含位运算符、溢出运算符、区间运算符等高级运算符。注意事项：溢出运算符防止整数溢出导致运行时错误，区间运算符用于表示范围。

# 高级运算符
// 溢出运算符
var unsignedOverflow = UInt8.max &+ 1 // 0
var signedOverflow = Int8.max &+ 1 // -128
// 位运算符
let initialBits: UInt8 = 0b00001111
let invertedBits = ~initialBits // 0b11110000
let firstSixBits: UInt8 = 0b11111100
let lastSixBits: UInt8 = 0b00111111
let middleFourBits = firstSixBits & lastSixBits // 0b00111100
// 区间运算符
for index in 1...5 {
    print("\(index) times 5 is \(index * 5)")
}
let names = ["Anna", "Alex", "Brian", "Jack"]
let range = 1..<3
for index in range {
    print(names[index])
}
// 空合运算符
var optionalString: String? = nil
let text = optionalString ?? "默认值" // "默认值"

17. 嵌套类型

Swift - 嵌套类型

特点：允许在一个类型内部定义另一个类型，提高代码的组织性和可读性。注意事项：嵌套类型可以是类、结构体或枚举，可以访问外部类型的属性和方法。

# 嵌套类型
// 嵌套枚举
struct BlackjackCard {
    // 嵌套枚举
    enum Suit {
        case spades, hearts, diamonds, clubs
        var description: String {
            switch self {
            case .spades:
                return "♠"
            case .hearts:
                return "♥"
            case .diamonds:
                return "♦"
            case .clubs:
                return "♣"
            }
        }
    }
    enum Rank {
        case two, three, four, five, six, seven, eight, nine, ten
        case jack, queen, king, ace
        struct Values {
            let first: Int
            let second: Int?
        }
        var values: Values {
            switch self {
            case .ace:
                return Values(first: 1, second: 11)
            case .jack, .queen, .king:
                return Values(first: 10, second: nil)
            default:
                return Values(first: 10, second: nil)
            }
        }
    }
    let rank: Rank
    let suit: Suit
    var description: String {
        return "The \(rank) of \(suit)"
    }
}
// 使用嵌套类型
let theAceOfSpades = BlackjackCard(rank: BlackjackCard.Rank.ace, suit: BlackjackCard.Suit.spades)

18. 自定义运算符

Swift - 自定义运算符

特点：允许创建自定义的运算符，包括前缀、中缀和后缀运算符。注意事项：自定义运算符需要先声明，然后实现，运算符的优先级和结合性可以自定义。

# 自定义运算符
// 自定义前缀运算符
prefix operator +++
struct Vector2D {
    var x = 0.0
    var y = 0.0
}
prefix func +++(vector: inout Vector2D) {
    vector.x += vector.x
    vector.y += vector.y
}
var toBeDoubled = Vector2D(x: 1.0, y: 4.0)
+++toBeDoubled // x: 2.0, y: 8.0
// 自定义中缀运算符
infix operator +-: AdditionPrecedence
extension Vector2D {
    static func +- (left: Vector2D, right: Vector2D) -> Vector2D {
        return Vector2D(x: left.x + right.x, y: left.y - right.y)
    }
}
let firstVector = Vector2D(x: 1.0, y: 2.0)
let secondVector = Vector2D(x: 3.0, y: 4.0)
let plusMinusVector = firstVector +- secondVector // x: 4.0, y: -2.0
// 自定义赋值运算符
infix operator +=-: AssignmentPrecedence
extension Vector2D {
    static func +=- (left: inout Vector2D, right: Vector2D) {
    left += left +- right
    }
}

19. 扩展与协议组合

Swift - 扩展与协议组合

特点：扩展允许向现有类型添加新功能，协议组合允许将多个协议组合成一个要求。注意事项：扩展可以添加方法、属性、下标和初始化器，协议组合使用&符号连接多个协议。

# 扩展与协议组合
// 扩展
extension Int {

Swift - 泛型约束

特点：泛型约束限制了泛型类型可以接受的具体类型，确保类型具有特定的功能。注意事项：可以约束泛型类型遵循特定的协议或继承自特定的类，也可以使用where子句添加更复杂的约束。

# 泛型约束
// 协议约束
func findIndexT: Equatable>(_ array: [T], _ valueToFind: T) -> Int? {
    for (index, value) in array.enumerated() {
        if value == valueToFind {
            return index
        }
    }
    return nil
}
let doubleIndex = findIndex([3.14159, 0.1, 0.25], 9.3) // nil
let stringIndex = findIndex(["Mike", "Malcolm", "Andrea"], "Andrea") // 2

protocol Container {
    associatedtype Item
    func contains(_ item: Item) -> Bool
}
func allItemsMatch<C1: Container, C2: Container>(_ someContainer: C1, _ anotherContainer: C2) -> Bool where C1.Item == C2.Item, C1.Item: Equatable {
    guard someContainer.count == anotherContainer.count else {
        return false
    }
    for i in 0..<someContainer.count {
        guard someContainer[i] == anotherContainer[i] else {
            return false
        }
    }
    return true
