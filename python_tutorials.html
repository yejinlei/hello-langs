19. 生成器表达式

Python - 惰性计算的迭代器创建方式

特点：惰性求值，节省内存，支持迭代，语法与列表推导式类似。注意事项：生成器只能迭代一次，迭代后为空。

# 基本生成器表达式
gen = (x*x for x in range(10))
print(gen) # at 0x...> print(list(gen)) # [0, 1, 4, 9, 16, 25, 36, 49, 64, 81]
# 生成器的惰性求值
gen = (x*x for x in range(10))
print(next(gen)) # 0
print(next(gen)) # 1
print(next(gen)) # 4
# 生成器在函数中的应用
def fibonacci(n):
    a, b = 0, 1
    for _ in range(n):
        yield a
        a, b = b, a + b

fib = fibonacci(5)
print(list(fib)) # [0, 1, 1, 2, 3]

20. 装饰器

Python - 增强函数或类的功能

特点：在不修改原函数代码的情况下增强函数功能，支持嵌套装饰器，支持带参数的装饰器。注意事项：装饰器执行顺序从下到上。

# 基本装饰器
def decorator(func):
    def wrapper(*args, **kwargs):
        print("函数执行前")
result = func(*args, **kwargs)
        print("函数执行后")
        return result
    return wrapper
@decorator
def add(a, b):
    return a + b
result = add(3, 5)
print(result) # 8
# 带参数的装饰器
def repeat(times):
    def decorator(func):
        def wrapper(*args, **kwargs):
            for _ in range(times):
                func(*args, **kwargs)
        return wrapper
    return decorator
@repeat(3)
def say_hello():
    print("Hello!")

say_hello() # 输出3次"Hello!"

21. 上下文管理器

Python - 管理资源的安全方式

特点：自动管理资源，确保资源正确释放，支持自定义上下文管理器，使用with语句。注意事项：上下文管理器必须实现__enter__和__exit__方法。

# 使用with语句管理文件资源

    f.write("Hello, Context Manager!")
# 自定义上下文管理器类
class MyContextManager:
    def __enter__(self):
        print("进入上下文")
        return "上下文资源"
    def __exit__(self, exc_type, exc_val, exc_tb):
        print("退出上下文")
with MyContextManager() as resource:
    print(f"使用资源: {resource}")
# 使用contextlib装饰器创建上下文管理器
from contextlib import contextmanager
@contextmanager
def my_context():
    print("进入上下文")
try:
        yield "上下文资源"
finally:
        print("退出上下文")
with my_context() as resource:
    print(f"使用资源: {resource}")

23. 迭代器

Python - 支持迭代的对象和迭代器协议

特点：所有序列类型都是可迭代的，迭代器支持__next__()方法，使用iter()函数获取迭代器，支持for循环自动迭代。注意事项：迭代器只能迭代一次，迭代器耗尽后无法重新使用。

# 可迭代对象和迭代器
# 可迭代对象 my_list = [1, 2, 3, 4, 5] # 获取迭代器 my_iter = iter(my_list)
# 使用next()函数迭代 print(next(my_iter)) # 1 print(next(my_iter)) # 2 print(next(my_iter)) # 3
# 自定义迭代器
class MyIterator:
    def __init__(self, max_value):
        self.max_value = max_value
        self.current = 0

    def __iter__(self):
        return self

    def __next__(self):
        if self.current < self.max_value:
            value = self.current
            self.current += 1
            return value
        else:
            raise StopIteration

for num in MyIterator(5):
    print(num) # 输出: 0 1 2 3 4

24. 生成器

Python - 使用yield关键字创建迭代器

特点：简洁的迭代器创建方式，惰性求值，节省内存，支持无限序列。注意事项：生成器函数调用返回生成器对象，每次调用yield会暂停执行，下次调用从暂停处继续。

# 基本生成器函数
def count_up_to(max_value): current = 0 while current < max_value: yield current current += 1
# 使用生成器 for num in count_up_to(5): print(num) # 输出: 0 1 2 3 4
# 无限生成器
def infinite_counter():
    num = 0
    while True:
        yield num
        num += 1

# 使用无限生成器
counter = infinite_counter()
print(next(counter)) # 0
print(next(counter)) # 1
print(next(counter)) # 2

25. 闭包

Python - 嵌套函数可以访问外部函数的变量

特点：可以访问和修改外部函数的变量，保存状态，支持数据封装，延迟执行。注意事项：闭包会延长外部函数变量的生命周期，可能导致内存泄漏。

# 基本闭包
def outer_function(message):
    def inner_function():
        return message
    return inner_function

greeting = outer_function("Hello, Closure!")
print(greeting()) # Hello, Closure!
# 闭包保存状态
def counter():
    count = 0
    def increment():
        nonlocal count
        count += 1
        return count
    return increment

my_counter = counter()
print(my_counter()) # 1
print(my_counter()) # 2
print(my_counter()) # 3

29. 枚举

    GREEN = 2
    BLUE = 3

    print(color) # 输出所有颜色枚举成员

# 二叉搜索树类
class BST:
    def __init__(self):
        self.root = None
    def insert(self, val):
        if not self.root:
            self.root = TreeNode(val)
            return
        self._insert(self.root, val)
    def _insert(self, node, val):
        if val < node.val:
            if not node.left:
                node.left = TreeNode(val)
            else:
                self._insert(node.left, val)
        elif val > node.val:
            if not node.right:
                node.right = TreeNode(val)
            else:
                self._insert(node.right, val)
    def search(self, val):
        return self._search(self.root, val)
    def _search(self, node, val):
        if not node or node.val == val:
            return node
        if val < node.val:
            return self._search(node.left, val)
        else:
            return self._search(node.right, val)

# 使用邻接表表示无向图
class Graph:
    def __init__(self):
        self.adj_list = {}

    def add_vertex(self, vertex):
        if vertex not in self.adj_list:
            self.adj_list[vertex] = []

    def add_edge(self, v1, v2):
        if v1 in self.adj_list and v2 in self.adj_list:
            self.adj_list[v1].append(v2)
            self.adj_list[v2].append(v1)

    def bfs(self, start):
        visited = set()
        queue = [start]
        visited.add(start)
        result = []
        while queue:
            vertex = queue.pop(0)
            result.append(vertex)
            for neighbor in self.adj_list[vertex]:
                if neighbor not in visited:
                    visited.add(neighbor)
                    queue.append(neighbor)
        return result

44. 冒泡排序

# 基本冒泡排序
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr
# 优化版冒泡排序（提前终止）
def bubble_sort_optimized(arr):
    n = len(arr)
    for i in range(n):
        swapped = False
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        if not swapped:
            break
    return arr

45. 快速排序

Python - 快速排序实现

特点：通过选择基准元素将数组分为两部分，递归排序。平均时间复杂度O(n log n)，最坏O(n²)，空间复杂度O(log n)。是实际应用中最常用的排序算法。

# 基本快速排序
def quick_sort(arr):
    if len(arr) <= 1:
        return arr
    pivot = arr[len(arr) // 2]
    left = [x for x in arr if x < pivot]
    middle = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    return quick_sort(left) + middle + quick_sort(right)
# 原地快速排序
def quick_sort_inplace(arr, low, high):

        pi = self._partition(arr, low, high)
        self.quick_sort_inplace(arr, low, pi - 1)
        self.quick_sort_inplace(arr, pi + 1, high)

46. 归并排序

Python - 归并排序实现

特点：采用分治法，将数组递归分割后合并。时间复杂度O(n log n)，空间复杂度O(n)。稳定排序，适用于大规模数据。

# 归并排序
def merge_sort(arr):
    if len(arr) <= 1:
        return arr
    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])
    return merge(left, right)
def merge(left, right):
    result = []
    i = j = 0
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1
    result.extend(left[i:])
    result.extend(right[j:])
    return result

47. 堆排序

Python - 堆排序实现

特点：利用堆数据结构进行排序。时间复杂度O(n log n)，空间复杂度O(1)。不稳定排序，适用于需要原地排序的场景。

# 堆排序
def heap_sort(arr):
    n = len(arr)

    # 构建最大堆
    for i in range(n // 2 - 1, -1, -1):
        heapify(arr, n, i)

    # 逐个提取元素
    for i in range(n - 1, 0, -1):
        arr[0], arr[i] = arr[i], arr[0]
        heapify(arr, i, 0)
    return arr
def heapify(arr, n, i):
    largest = i
    left = 2 * i + 1
    right = 2 * i + 2

    if left < n and arr[left] > arr[largest]:
        largest = left
    if right < n and arr[right] > arr[largest]:
        largest = right

    if largest != i:
        arr[i], arr[largest] = arr[largest], arr[i]
        heapify(arr, n, largest)

49. 二分搜索

Python - 二分搜索实现

特点：在有序数组中通过比较中间元素来缩小搜索范围。时间复杂度O(log n)，空间复杂度O(1)。适用于有序数组的大规模搜索。

# 迭代版二分搜索
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = left + (right - left) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1
# 递归版二分搜索
def binary_search_recursive(arr, target, left=0, right=None):
    if right is None:
        right = len(arr) - 1
    if left > right:
        return -1
    mid = left + (right - left) // 2
    if arr[mid] == target:
        return mid
    elif arr[mid] < target:
        return binary_search_recursive(arr, target, mid + 1, right)
    else:
        return binary_search_recursive(arr, target, left, mid - 1)

50. 动态规划

Python - 动态规划实现

特点：通过将复杂问题分解为子问题并存储结果来避免重复计算。适用于最优化问题，如斐波那契数列、背包问题等。

# 斐波那契数列（动态规划）
def fibonacci_dp(n):
    if n <= 1:
        return n
    dp = [0] * (n + 1)
    dp[0], dp[1] = 0, 1
    for i in range(2, n + 1):
        dp[i] = dp[i - 1] + dp[i - 2]
    return dp[n]
# 背包问题（0-1背包）
def knapsack_01(weights, values, capacity):
    n = len(weights)
    dp = [[0 for _ in range(capacity + 1)] for _ in range(n + 1)]
    for i in range(1, n + 1):
        for w in range(1, capacity + 1):
            if weights[i - 1] <= w:
                dp[i][w] = max(dp[i - 1][w], values[i - 1] + dp[i - 1][w - weights[i - 1]])
            else:
                dp[i][w] = dp[i - 1][w]
    return dp[n][capacity]

51. 贪心算法

Python - 贪心算法实现

特点：在每一步选择中都采取当前状态下最优的选择，期望通过局部最优解得到全局最优解。适用于活动选择、哈夫曼编码等。

# 活动选择问题
def activity_selection(start, end):
    n = len(start)
    activities = list(zip(start, end))
    activities.sort(key=lambda x: x[1])
    result = []
    last_end = -float('inf')
    for s, e in activities:
        if s >= last_end:
            result.append((s, e))
            last_end = e
    return result
# 硬币找零问题
def coin_change(coins, amount):
    coins.sort(reverse=True)
    result = []
    for coin in coins:
        while amount >= coin:
            result.append(coin)
            amount -= coin
    return result

52. 回溯算法

Python - 回溯算法实现

特点：通过探索所有可能的候选解来找出所有解，如果发现当前路径不可能得到解就回溯。适用于N皇后、数独、全排列等。

# 全排列问题
def permute(nums):
    result = []
    used = [False] * len(nums)
    backtrack(nums, [], used, result)
    return result
def backtrack(nums, path, used, result):
    if len(path) == len(nums):
        result.append(path[:])
        return
    for i in range(len(nums)):
        if used[i]:
            continue
        used[i] = True
        path.append(nums[i])
        backtrack(nums, path, used, result)
        path.pop()
        used[i] = False

53. 图算法

Python - 图算法实现

特点：图算法用于处理图结构数据，包括最短路径、最小生成树、拓扑排序等。适用于网络分析、路径规划等场景。

# Dijkstra最短路径算法
import heapq
def dijkstra(graph, start):
    distances = {node: float('inf') for node in graph}
    distances[start] = 0
    pq = [(0, start)]
    visited = set()
    while pq:
        current_dist, current_node = heapq.heappop(pq)
        if current_node in visited:
            continue
        visited.add(current_node)
        for neighbor, weight in graph[current_node].items():
            if neighbor in visited:
                continue
            distance = current_dist + weight
            if distance < distances[neighbor]:
                distances[neighbor] = distance
                heapq.heappush(pq, (distance, neighbor))
    return distances
