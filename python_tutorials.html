19. 生成器表达式

Python - 惰性计算的迭代器创建方式

# 基本生成器表达式

# 生成器的惰性求值
gen = (x*x for x in range(10))

# 生成器在函数中的应用
def fibonacci(n):
    a, b = 0, 1
    for _ in range(n):
        yield a
        a, b = b, a + b

fib = fibonacci(5)
print(list(fib)) # [0, 1, 1, 2, 3]

20. 装饰器

Python - 增强函数或类的功能

特点：在不修改原函数代码的情况下增强函数功能，支持嵌套装饰器，支持带参数的装饰器。注意事项：装饰器执行顺序从下到上。

# 基本装饰器
def decorator(func):
    def wrapper(*args, **kwargs):
        print("函数执行前")
result = func(*args, **kwargs)
        print("函数执行后")
        return result
    return wrapper
@decorator
def add(a, b):
    return a + b
result = add(3, 5)
print(result) # 8
# 带参数的装饰器
def repeat(times):
    def decorator(func):
        def wrapper(*args, **kwargs):
            for _ in range(times):
                func(*args, **kwargs)
        return wrapper
    return decorator
@repeat(3)
def say_hello():
    print("Hello!")

say_hello() # 输出3次"Hello!"

21. 上下文管理器

Python - 管理资源的安全方式

特点：自动管理资源，确保资源正确释放，支持自定义上下文管理器，使用with语句。注意事项：上下文管理器必须实现__enter__和__exit__方法。

    f.write("Hello, Context Manager!")
# 自定义上下文管理器类
class MyContextManager:
    def __enter__(self):
        print("进入上下文")
        return "上下文资源"
    def __exit__(self, exc_type, exc_val, exc_tb):
        print("退出上下文")
with MyContextManager() as resource:
    print(f"使用资源: {resource}")
# 使用contextlib装饰器创建上下文管理器
from contextlib import contextmanager
@contextmanager
def my_context():
    print("进入上下文")
try:
        yield "上下文资源"
finally:
        print("退出上下文")
with my_context() as resource:
    print(f"使用资源: {resource}")

23. 迭代器

Python - 支持迭代的对象和迭代器协议

特点：所有序列类型都是可迭代的，迭代器支持__next__()方法，使用iter()函数获取迭代器，支持for循环自动迭代。注意事项：迭代器只能迭代一次，迭代器耗尽后无法重新使用。

# 可迭代对象和迭代器
# 可迭代对象 my_list = [1, 2, 3, 4, 5] # 获取迭代器 my_iter = iter(my_list)
# 使用next()函数迭代 print(next(my_iter)) # 1 print(next(my_iter)) # 2 print(next(my_iter)) # 3
# 自定义迭代器
class MyIterator:
    def __init__(self, max_value):
        self.max_value = max_value
        self.current = 0

    def __iter__(self):
        return self

    def __next__(self):
        if self.current < self.max_value:
            value = self.current
            self.current += 1
            return value
        else:
            raise StopIteration

for num in MyIterator(5):
    print(num) # 输出: 0 1 2 3 4

24. 生成器

Python - 使用yield关键字创建迭代器

特点：简洁的迭代器创建方式，惰性求值，节省内存，支持无限序列。注意事项：生成器函数调用返回生成器对象，每次调用yield会暂停执行，下次调用从暂停处继续。

# 基本生成器函数
def count_up_to(max_value): current = 0 while current < max_value: yield current current += 1
# 使用生成器 for num in count_up_to(5): print(num) # 输出: 0 1 2 3 4
# 无限生成器
def infinite_counter():
    num = 0
    while True:
        yield num
        num += 1

# 使用无限生成器
counter = infinite_counter()
print(next(counter)) # 0
print(next(counter)) # 1
print(next(counter)) # 2

25. 闭包

Python - 嵌套函数可以访问外部函数的变量

特点：可以访问和修改外部函数的变量，保存状态，支持数据封装，延迟执行。注意事项：闭包会延长外部函数变量的生命周期，可能导致内存泄漏。

# 基本闭包
def outer_function(message):
    def inner_function():
        return message
    return inner_function

greeting = outer_function("Hello, Closure!")
print(greeting()) # Hello, Closure!
# 闭包保存状态
def counter():
    count = 0
    def increment():
        nonlocal count
        count += 1
        return count
    return increment

my_counter = counter()
print(my_counter()) # 1
print(my_counter()) # 2
print(my_counter()) # 3

31. 异步编程

Python - 支持异步编程，提高I/O密集型任务的性能

# 并发运行多个异步任务
import asyncio
async def task(name, delay):     print(f"任务{name}开始")
await asyncio.sleep(delay)
    print(f"任务{name}完成")
return f"结果{name}"
async def main():
    # 创建多个任务
    tasks = [
        task("A", 1),
        task("B", 2),
        task("C", 1)
    ]
    # 并发运行所有任务
    results = await asyncio.gather(*tasks)
    print(results)
asyncio.run(main())

32. 高级类型

Python - 高级类型提示功能，增强代码的类型安全性

特点：支持泛型，支持联合类型，支持可选类型，支持字面量类型，支持协议类型。注意事项：需要Python 3.8+版本，结合类型检查工具如mypy使用效果更佳。

# 联合类型和可选类型
from typing import Union, Optional
def process_value(value: Union[int, str]) -> str:
    return str(value)
def get_name(person: dict) -> Optional[str]:
    return person.get("name")
# 泛型类型
from typing import List, Dict, Generic, TypeVar
T = TypeVar('T')
class Box(Generic[T]):     def __init__(self, content: T):
        self.content = content
    def get_content(self) -> T:
        return self.content
# 使用泛型类
int_box = Box(42)
str_box = Box("Hello")

print(int_box.get_content()) # 42
print(str_box.get_content()) # Hello

33. 元编程

Python - 编写能够操作代码的代码

特点：支持动态属性，支持装饰器，支持元类，支持反射，支持动态导入。注意事项：元编程可能使代码变得复杂，降低可读性，应谨慎使用。

# 动态属性和方法
class DynamicClass:     def __init__(self):
        self.dynamic_attr = 42
obj = DynamicClass()

# 动态添加属性
obj.new_attr = "动态属性"

# 动态添加方法
def dynamic_method(self):
    return "动态方法调用"
import types
obj.dynamic_method = types.MethodType(dynamic_method, obj)

print(obj.dynamic_attr) # 42
print(obj.new_attr) # 动态属性
print(obj.dynamic_method()) # 动态方法调用
# 元类
class MetaClass(type):     def __new__(cls, name, bases, attrs):         # 在创建类之前修改属性
        attrs['created_by_meta'] = True         return super().__new__(cls, name, bases, attrs)
class MyClass(metaclass=MetaClass):     pass
print(MyClass.created_by_meta) # True
