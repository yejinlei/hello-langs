1. Shell基础

什么是Shell？

Shell是一种命令行解释器，它提供了用户与操作系统内核进行交互的界面。Bash (Bourne Again Shell) 是最常用的Shell之一。

基本命令

echo "Hello, World!"  # 输出文本
pwd                    # 显示当前工作目录
ls                     # 列出目录内容
cd /path/to/directory   # 切换目录
mkdir new_directory     # 创建新目录
rm file.txt            # 删除文件

命令格式

Shell命令的基本格式：

command [options] [arguments]

示例：

ls -l -a  # 长格式列出所有文件（包括隐藏文件）
cp file1.txt file2.txt  # 复制文件

目录操作

3. 文本处理命令

基本文本处理

cat file1.txt file2.txt > combined.txt  # 合并文件
sort file.txt                    # 排序文件内容
uniq file.txt                    # 去除重复行
wc -l file.txt                # 统计行数
grep "pattern" file.txt           # 查找匹配模式的行
sed 's/old/new/g' file.txt      # 替换文本
awk '{print $1}' file.txt        # 打印第一列

更复杂的文本处理

grep -i -r "error" /var/log/  # 递归查找错误日志
sed 's/^/# /' file.txt               # 在每行前添加注释
awk '{sum += $2} END {print sum}' file.txt  # 计算第二列总和
cut -d"," -f2 file.csv         # 提取CSV文件的第二列

4. 变量和赋值

变量定义和使用

name="John"        # 定义变量（注意等号前后不能有空格）
echo $name         # 使用变量
echo "My name is $name"  # 在字符串中使用变量
echo "My name is ${name}s" # 变量名扩展

命令替换

files=`ls`         # 使用反引号进行命令替换
files=$(ls)        # 使用$()进行命令替换（推荐）
echo $files

date=$(date +%Y-%m-%d)
echo "Today is $date"

5. 字符串操作

字符串长度

str="Hello World"
echo ${#str}  # 输出：11

字符串截取

str="Hello World"
echo ${str:6}    # 从位置6开始截取：World
echo ${str:0:5}  # 从位置0开始截取5个字符：Hello
echo ${str: -5}   # 截取最后5个字符：World

字符串替换

str="Hello World"
echo ${str/World/PHP}   # 替换第一个匹配：Hello PHP
echo ${str//l/L}      # 替换所有匹配：HeLLo WorLd
echo ${str/#Hello/Hi}   # 替换开头：Hi World
echo ${str/%World/Shell} # 替换结尾：Hello Shell

6. 数组操作

数组定义和访问

fruits=("apple" "banana" "orange")  # 定义数组
echo ${fruits[0]}         # 访问第一个元素：apple
echo ${fruits[1]}         # 访问第二个元素：banana
echo ${fruits[@]}         # 访问所有元素
echo ${fruits[*]}         # 访问所有元素

数组操作

fruits=("apple" "banana")
fruits+=("orange")           # 添加元素
fruits[2]="grape"          # 修改元素

echo ${#fruits[@]}       # 数组长度

for fruit in ${fruits[@]}; do
    echo $fruit
done

关联数组

declare -A person  # 声明关联数组
person["name"]="John"
person["age"]=30
person["city"]="New York"

echo ${person["name"]}  # 访问：John

echo ${!person[@]}      # 所有键
echo ${person[@]}       # 所有值

7. 条件语句

if语句

if [ $a -gt 10 ]; then
    echo "a is greater than 10"
elif [ $a -lt 10 ]; then
    echo "a is less than 10"
else
    echo "a is equal to 10"
fi

字符串条件

name="John"

if [ "$name" == "John" ]; then
    echo "Hello John!"
fi

if [ -z "$empty" ]; then
    echo "String is empty"
fi

if [ -n "$name" ]; then
    echo "String is not empty"
fi

文件条件

if [ -f "file.txt" ]; then
    echo "File exists"
fi

if [ -d "directory" ]; then
    echo "Directory exists"
fi

if [ -r "file.txt" ]; then
    echo "File is readable"
fi

if [ -w "file.txt" ]; then
    echo "File is writable"
fi

if [ -x "script.sh" ]; then
    echo "Script is executable"
fi

8. 循环语句

for循环

for i in 1 2 3 4 5; do
    echo "Number: $i"
done

for i in {1..10}; do
    echo "Number: $i"
done

for file in *.txt; do
    echo "File: $file"
done

while循环

i=1
while [ $i -le 5 ]; do
    echo "Number: $i"
    i=$((i + 1))
done

cat file.txt | while read line; do
    echo "Line: $line"
done

until循环

i=1
until [ $i -gt 5 ]; do
    echo "Number: $i"
    i=$((i + 1))
done

9. 函数

函数定义和调用

function greet {
    name=$1
    echo "Hello, $name!"
}

greet "John"  # 调用函数

add () {
    sum=$(($1 + $2))
    echo $sum
}

result=$(add 5 3)
echo "Result: $result"

返回值和作用域

multiply () {
    product=$(($1 * $2))
    return $product
}

multiply 4 5
result=$?  # 获取返回值
echo "Result: $result"

global_var="global"

local_example () {
    local local_var="local"
    global_var="modified"  # 修改全局变量
    echo "Local: $local_var"
}

local_example
echo "Global: $global_var"

12. 正则表达式

基本正则表达式

grep "^a" file.txt          # 匹配以a开头的行
grep "z$" file.txt          # 匹配以z结尾的行
grep "[0-9]" file.txt        # 匹配包含数字的行
grep "[a-zA-Z]" file.txt     # 匹配包含字母的行
grep "^[A-Z]" file.txt       # 匹配以大写字母开头的行

扩展正则表达式

grep -E "a{2,3}" file.txt     # 匹配2-3个连续的a
grep -E "(cat|dog)" file.txt    # 匹配cat或dog
grep -E "[0-9]{3}-[0-9]{3}-[0-9]{4}" file.txt  # 匹配电话号码格式

16. 权限管理

查看权限

ls -l file.txt           # 查看文件权限
ls -ld directory/        # 查看目录权限

修改权限

chmod +x script.sh         # 赋予执行权限
chmod -x script.sh         # 移除执行权限
chmod 644 file.txt          # 设置权限为rw-r--r--
chmod 755 directory/       # 设置目录权限为rwxr-xr-x

# 使用符号表示法
chmod u+x script.sh        # 给所有者添加执行权限
chmod g+w file.txt         # 给组添加写权限
chmod o-r file.txt         # 移除其他人的读权限

修改所有者

chown user:group file.txt      # 修改所有者和组
chown user file.txt            # 只修改所有者
chgrp group file.txt           # 只修改组

chown -R user:group directory/ # 递归修改目录

18. 压缩和归档

tar命令

tar -cvf archive.tar directory/  # 创建归档
tar -xvf archive.tar          # 解压缩归档
tar -tvf archive.tar          # 查看归档内容

# 压缩为gzip格式
tar -czvf archive.tar.gz directory/
tar -xzvf archive.tar.gz

# 压缩为bzip2格式
tar -cjvf archive.tar.bz2 directory/
tar -xjvf archive.tar.bz2

其他压缩命令

gzip file.txt                # 压缩为.gz文件
gunzip file.txt.gz            # 解压缩.gz文件

bzip2 file.txt               # 压缩为.bz2文件
bunzip2 file.txt.bz2          # 解压缩.bz2文件

zip archive.zip file1.txt file2.txt  # 创建zip压缩文件
unzip archive.zip                 # 解压缩zip文件

19. 高级主题

命令别名

alias ll='ls -la'              # 创建别名
alias gs='git status'         # Git状态别名
unalias ll                   # 移除别名

# 永久别名（添加到~/.bashrc或~/.bash_profile）
echo 'alias ll="ls -la"' >> ~/.bashrc

Here Documents

cat >> file.txt <<- EOF
这是一个
多行文本
EOF

正则表达式高级用法

# 使用sed进行复杂替换
sed '1,5d' file.txt            # 删除前5行
sed '/pattern/d' file.txt      # 删除包含pattern的行
sed 's/^/    /' file.txt        # 每行前添加4个空格

# 使用awk进行高级文本处理
awk 'NR % 2 == 0' file.txt     # 打印偶数行
awk '$1 > 100' file.txt      # 打印第一列大于100的行
awk '{sum += $2} END {print "Total: " sum}' file.txt
