1. Erlang 简介

Erlang - 专为高并发和容错设计的编程语言

Erlang是一种函数式编程语言，最初由爱立信公司开发，用于构建高可用性、低延迟的电信系统。它以其卓越的并发处理能力、分布式计算支持和内置的容错机制而闻名，广泛应用于需要高可靠性的系统开发。

Erlang 语言特点
  • 轻量级进程模型，支持数百万级并发
  • 内置分布式计算支持
  • 健壮的容错机制
  • 热代码升级能力
  • 函数式编程范式
  • 动态类型系统
  • 垃圾回收机制
  • 模式匹配和递归

3. Erlang 函数和模块

函数定义和调用
% 模块定义
-module(math_functions).
-export([add/2, multiply/2, factorial/1]).

% 函数定义
add(A, B) ->
    A + B.

multiply(A, B) ->
    A * B.

% 递归函数
factorial(0) -> 1;
factorial(N) when N > 0 ->
    N * factorial(N-1).

% 调用函数
% math_functions:add(5, 3). % 8
% math_functions:factorial(5). % 120
模式匹配
% 列表模式匹配
head([H | _]) -> H;
head([]) -> empty.

tail([_ | T]) -> T;
tail([]) -> empty.

% 元组模式匹配
area({circle, R}) ->
    math:pi() * R * R;
area({rectangle, W, H}) ->
    W * H;
area({triangle, B, H}) ->
    0.5 * B * H;
area(_) ->
    {error, invalid_shape}.

% 函数参数模式匹配
greet({person, Name, _Age}) ->
    io:fwrite("Hello, ~s!~n", [Name]);
greet(_) ->
    io:fwrite("Hello, stranger!~n").

4. Erlang 控制流

条件表达式
% if 表达式
max(X, Y) ->
    if
        X > Y -> X;
        true -> Y
    end.

% case 表达式
classify_age(Age) ->
    case Age of
        N when N < 18 -> minor;
        N when N < 65 -> adult;
        _ -> senior
    end.

% 守卫表达式
positive(X) when X > 0 -> true;
positive(_) -> false.

% 多重守卫
divide(X, Y) when Y /= 0 -> {ok, X / Y};
divide(_, 0) -> {error, division_by_zero}.
列表操作
% 列表推导式
EvenNumbers = [X || X <- [1, 2, 3, 4, 5], X rem 2 == 0]. % [2, 4]

% 带条件的列表推导式
SquaresOfEven = [X*X || X <- [1, 2, 3, 4, 5], X rem 2 == 0]. % [4, 16]

% 多重生成器
Pairs = [{X, Y} || X <- [1, 2, 3], Y <- [a, b]]. % [{1,a}, {1,b}, {2,a}, {2,b}, {3,a}, {3,b}]

% 内置列表函数
lists:reverse([1, 2, 3]). % [3, 2, 1]
lists:sort([3, 1, 2]). % [1, 2, 3]
lists:sum([1, 2, 3]). % 6
lists:map(fun(X) -> X*2 end, [1, 2, 3]). % [2, 4, 6]
lists:filter(fun(X) -> X rem 2 == 0 end, [1, 2, 3, 4]). % [2, 4]

5. Erlang 并发编程

进程创建和消息传递
% 进程创建
-module(process_example).
-export([start/0, worker/0]).

start() ->
    % 创建新进程
    Pid = spawn(process_example, worker, []),
    % 发送消息
    Pid ! {hello, "world"},
    Pid ! stop.

worker() ->
    receive
        {hello, Msg} ->
            io:fwrite("Received: ~s~n", [Msg]),
            worker();
        stop ->
            io:fwrite("Stopping worker~n");
        _ ->
            io:fwrite("Unknown message~n"),
            worker()
    end.
进程链接和监控
% 进程链接
-module(link_example).
-export([start/0, worker/0]).

start() ->
    % 链接进程
    spawn_link(link_example, worker, []),
    io:fwrite("Main process continues~n").

worker() ->
    io:fwrite("Worker started~n"),
    % 故意崩溃
    1 / 0.

% 进程监控
-module(monitor_example).
-export([start/0, worker/0]).

start() ->
    % 监控进程
    Pid = spawn(monitor_example, worker, []),
    Ref = monitor(process, Pid),
    receive
        {'DOWN', Ref, process, Pid, Reason} ->
            io:fwrite("Worker died: ~p~n", [Reason])
    end.

worker() ->
    io:fwrite("Worker started~n"),
    % 故意崩溃
    1 / 0.

6. Erlang OTP 行为模式

GenServer 行为模式
% 简单的 GenServer 实现
-module(counter_server).
-behaviour(gen_server).

% API 函数
export([start_link/0, increment/1, decrement/1, get_value/1]).

% GenServer 回调函数
export([init/1, handle_call/3, handle_cast/2, handle_info/2, terminate/2, code_change/3]).

% 状态
-record(state, {count = 0}).

% API 实现
start_link() ->
    gen_server:start_link(?MODULE, [], []).

increment(Pid) ->
    gen_server:cast(Pid, increment).

decrement(Pid) ->
    gen_server:cast(Pid, decrement).

get_value(Pid) ->
    gen_server:call(Pid, get_value).

% 回调函数实现
init([]) ->
    {ok, #state{}}.

handle_call(get_value, _From, State) ->
    {reply, State#state.count, State}.

handle_cast(increment, State) ->
    NewCount = State#state.count + 1,
    {noreply, State#state{count = NewCount}};
handle_cast(decrement, State) ->
    NewCount = State#state.count - 1,
    {noreply, State#state{count = NewCount}}.

handle_info(_Info, State) ->
    {noreply, State}.

terminate(_Reason, _State) ->
    ok.

code_change(_OldVsn, State, _Extra) ->
    {ok, State}.
