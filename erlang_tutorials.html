1. Erlang 简介

Erlang 概述

Erlang是一种通用的、并发的、函数式编程语言，由爱立信公司开发，专为构建高可用、低延迟的分布式系统而设计。

历史背景
  • 1986年由爱立信公司开发
  • 旨在解决电信系统的可靠性问题
  • 2007年发布开源版本
  • 广泛应用于电信、金融等领域
核心特性
  • 并发处理模型
  • 容错机制
  • 热代码升级
  • 分布式计算
  • 函数式编程

5. Erlang 错误处理

错误处理机制
% try-catch 表达式
safe_divide(A, B) ->
try
  A / B
catch
  error:badarith -> "Division by zero";
  _:_ -> "Unknown error"
end.

8. Erlang OTP框架

GenServer示例
% 定义GenServer模块
-module(my_server).
-behaviour(gen_server).
-export([start_link(), increment(), get_count()]).
-export([init(1), handle_call(3), handle_cast(2)]).

-record(state, {count = 0}).

start_link() ->
  gen_server:start_link({local, ?MODULE}, ?MODULE, [], []).

increment() ->
  gen_server:cast(?MODULE, increment).

get_count() ->
  gen_server:call(?MODULE, get_count).

init([]) ->
  {ok, #state{}}.

handle_call(get_count, _From, State) ->
  {reply, State#state.count, State}.

handle_cast(increment, State) ->
  NewCount = State#state.count + 1,
  {noreply, State#state{count = NewCount}}.

9. Erlang 分布式编程

节点间通信
% 分布式节点通信
-module(distributed_example).
-export([start_node(1), send_message(2)]).

start_node(NodeName) ->
  Pid = spawn(fun() -> receive_loop() end),
  register(receiver, Pid),
  io:fwrite("Receiver registered on node: ~p~n", [node()]).

send_message(TargetNode, Message) ->
  {receiver, TargetNode} ! Message,
  io:fwrite("Sent message to node: ~p~n", [TargetNode]).

receive_loop() ->
  receive
    Message ->
      io:fwrite("Received message on node ~p: ~p~n", [node(), Message]),
      receive_loop()
  end.
