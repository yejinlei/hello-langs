1. SQL基础

什么是SQL？

SQL（Structured Query Language）是用于管理关系型数据库的标准语言。它允许用户执行各种数据库操作，包括查询数据、插入数据、更新数据和删除数据。

SQL的类型

SQL主要分为以下几类：

  • DQL（数据查询语言）：用于查询数据，主要是SELECT语句
  • DML（数据操作语言）：用于操作数据，包括INSERT、UPDATE、DELETE
  • DDL（数据定义语言）：用于定义数据库结构，包括CREATE、ALTER、DROP
  • DCL（数据控制语言）：用于控制数据库权限，包括GRANT、REVOKE
  • TCL（事务控制语言）：用于控制事务，包括COMMIT、ROLLBACK、SAVEPOINT
-- 这是一个SQL注释
SELECT column1, column2
FROM table_name
WHERE condition;

字符串类型

SET -- 集合类型

3. SELECT语句

基本SELECT语句

SELECT语句用于从数据库表中检索数据。

-- 选择所有列
SELECT *
FROM employees;

-- 选择特定列
SELECT first_name, last_name, salary
FROM employees;

-- 使用别名
SELECT first_name AS "名", last_name AS "姓"
FROM employees;

DISTINCT关键字

DISTINCT关键字用于返回唯一不同的值。

-- 选择唯一的部门
SELECT DISTINCT department
FROM employees;

LIMIT子句

LIMIT子句用于限制查询结果的数量。

-- 限制结果为5行
SELECT *
FROM employees
LIMIT 5;

-- 从第3行开始，返回5行
SELECT *
FROM employees
LIMIT 2, 5;

4. WHERE子句

基本WHERE条件

WHERE子句用于过滤记录，只返回满足指定条件的记录。

-- 选择工资大于5000的员工
SELECT *
FROM employees
WHERE salary > 5000;

-- 选择部门为'IT'的员工
SELECT first_name, last_name
FROM employees
WHERE department = 'IT';

逻辑运算符

可以使用逻辑运算符AND、OR、NOT来组合多个条件。

-- AND运算符
SELECT *
FROM employees
WHERE department = 'IT'
AND salary > 6000;

-- OR运算符
SELECT *
FROM employees
WHERE department = 'IT'
OR department = 'Finance';

-- NOT运算符
SELECT *
FROM employees
WHERE NOT department = 'HR';

比较运算符

-- 等于
column = 100

-- 不等于
column != 100
column <> 100

-- 大于
column > 100

-- 大于等于
column >= 100

-- 小于
column < 100

-- 小于等于
column <= 100

特殊运算符

-- BETWEEN运算符
SELECT *
FROM employees
WHERE salary BETWEEN 5000 AND 8000;

-- IN运算符
SELECT *
FROM employees
WHERE department IN ('IT', 'Finance', 'Marketing');

-- LIKE运算符
SELECT *
FROM employees
WHERE first_name LIKE 'J%'; -- 以J开头

SELECT *
FROM employees
WHERE last_name LIKE '%son'; -- 以son结尾

SELECT *
FROM employees
WHERE email LIKE '%@gmail.com'; -- 包含@gmail.com

-- IS NULL运算符
SELECT *
FROM employees
WHERE manager_id IS NULL;

-- IS NOT NULL运算符
SELECT *
FROM employees
WHERE manager_id IS NOT NULL;

5. 排序和分组

ORDER BY排序

ORDER BY子句用于对结果集进行排序。

-- 升序排序（默认）
SELECT first_name, last_name, salary
FROM employees
ORDER BY salary;

-- 降序排序
SELECT first_name, last_name, salary
FROM employees
ORDER BY salary DESC;

-- 多列排序
SELECT department, first_name, salary
FROM employees
ORDER BY department ASC, salary DESC;

GROUP BY分组

GROUP BY子句用于结合聚合函数，根据一个或多个列对结果集进行分组。

-- 计算每个部门的平均工资
SELECT department, AVG(salary) AS avg_salary
FROM employees
GROUP BY department;

-- 计算每个部门的员工数量
SELECT department, COUNT(*) AS employee_count
FROM employees
GROUP BY department;

-- 计算每个部门的最高和最低工资
SELECT department,
MAX(salary) AS max_salary,
MIN(salary) AS min_salary
FROM employees
GROUP BY department;

HAVING子句

HAVING子句用于过滤分组后的结果。

-- 选择平均工资大于6000的部门
SELECT department, AVG(salary) AS avg_salary
FROM employees
GROUP BY department
HAVING AVG(salary) > 6000;

-- 选择员工数量大于5的部门
SELECT department, COUNT(*) AS employee_count
FROM employees
GROUP BY department
HAVING COUNT(*) > 5;

6. JOIN操作

INNER JOIN

INNER JOIN返回两个表中匹配的行。

-- 连接employees和departments表
SELECT e.first_name, e.last_name, d.department_name
FROM employees AS e
INNER JOIN departments AS d
ON e.department_id = d.department_id;

LEFT JOIN

LEFT JOIN返回左表中的所有行，以及右表中匹配的行。

-- 左连接employees和departments表
SELECT e.first_name, e.last_name, d.department_name
FROM employees AS e
LEFT JOIN departments AS d
ON e.department_id = d.department_id;

RIGHT JOIN

RIGHT JOIN返回右表中的所有行，以及左表中匹配的行。

-- 右连接employees和departments表
SELECT e.first_name, e.last_name, d.department_name
FROM employees AS e
RIGHT JOIN departments AS d
ON e.department_id = d.department_id;

FULL JOIN

FULL JOIN返回左表和右表中的所有行。

-- 全连接employees和departments表
SELECT e.first_name, e.last_name, d.department_name
FROM employees AS e
FULL JOIN departments AS d
ON e.department_id = d.department_id;

CROSS JOIN

CROSS JOIN返回两个表的笛卡尔积。

-- 交叉连接employees和departments表
SELECT e.first_name, d.department_name
FROM employees AS e
CROSS JOIN departments AS d;

SELF JOIN

SELF JOIN是指表与其自身连接。

-- 查找每个员工的经理
SELECT e.first_name AS employee_name,
m.first_name AS manager_name
FROM employees AS e
INNER JOIN employees AS m
ON e.manager_id = m.employee_id;

7. 子查询

基本子查询

子查询是嵌套在其他SQL语句中的查询。

-- 查找工资高于平均工资的员工
SELECT first_name, last_name, salary
FROM employees
WHERE salary > (SELECT AVG(salary) FROM employees);

-- 查找与John在同一部门的员工
SELECT first_name, last_name
FROM employees
WHERE department_id = (
SELECT department_id
FROM employees
WHERE first_name = 'John'
);

IN子查询

IN子查询用于检查值是否在子查询返回的结果集中。

-- 查找在IT或Finance部门工作的员工
SELECT first_name, last_name
FROM employees
WHERE department_id IN (
SELECT department_id
FROM departments
WHERE department_name IN ('IT', 'Finance')
);

EXISTS子查询

EXISTS子查询用于检查子查询是否返回任何行。

-- 查找有员工的部门
SELECT department_name
FROM departments d
WHERE EXISTS (
SELECT *
FROM employees e
WHERE e.department_id = d.department_id
);

ANY和ALL子查询

ANY和ALL运算符用于将单个值与子查询返回的多个值进行比较。

-- 查找工资高于任何IT部门员工工资的员工
SELECT first_name, last_name, salary
FROM employees
WHERE salary > ANY (
SELECT salary
FROM employees
WHERE department_id = 10 -- IT部门ID
);

-- 查找工资高于所有IT部门员工工资的员工
SELECT first_name, last_name, salary
FROM employees
WHERE salary > ALL (
SELECT salary
FROM employees
WHERE department_id = 10 -- IT部门ID
);

FROM子句中的子查询

子查询可以出现在FROM子句中，作为临时表。

-- 查找每个部门的平均工资和员工数量
SELECT dept_avg.department_name, dept_avg.avg_salary, dept_count.employee_count
FROM
(SELECT d.department_name, AVG(e.salary) AS avg_salary
FROM employees e
INNER JOIN departments d ON e.department_id = d.department_id
GROUP BY d.department_name
) dept_avg
INNER JOIN
(SELECT d.department_name, COUNT(*) AS employee_count
FROM employees e
INNER JOIN departments d ON e.department_id = d.department_id
GROUP BY d.department_name
) dept_count
ON dept_avg.department_name = dept_count.department_name;

8. 插入数据

基本INSERT语句

INSERT语句用于向表中插入新记录。

-- 插入完整记录
INSERT INTO employees (first_name, last_name, email, salary, department_id)
VALUES ('John', 'Doe', 'john.doe@example.com', 6000, 10);

-- 插入部分字段
INSERT INTO employees (first_name, last_name, email)
VALUES ('Jane', 'Smith', 'jane.smith@example.com');

插入多条记录

可以使用一条INSERT语句插入多条记录。

INSERT INTO employees (first_name, last_name, email, salary)
VALUES
('Bob', 'Johnson', 'bob.johnson@example.com', 5500),
('Alice', 'Williams', 'alice.williams@example.com', 6200),
('Charlie', 'Brown', 'charlie.brown@example.com', 5800);

从另一个表插入数据

可以从一个表向另一个表插入数据。

-- 创建备份表
CREATE TABLE employees_backup AS
SELECT * FROM employees WHERE department_id = 10;

-- 插入特定数据
INSERT INTO employees_backup (first_name, last_name, email)
SELECT first_name, last_name, email
FROM employees
WHERE department_id = 20;

9. 更新数据

基本UPDATE语句

UPDATE语句用于修改表中的现有记录。

-- 更新单个记录
UPDATE employees
SET salary = 6500
WHERE employee_id = 1;

-- 更新多个字段
UPDATE employees
SET salary = 7000,
email = 'john.doe@company.com'
WHERE first_name = 'John' AND last_name = 'Doe';

更新多条记录

可以使用WHERE子句来过滤需要更新的记录。

-- 更新所有IT部门员工的工资
UPDATE employees
SET salary = salary * 1.1
WHERE department_id = 10;

使用子查询更新

可以在UPDATE语句中使用子查询来基于其他表的数据更新记录。

-- 更新与John在同一部门的员工工资
UPDATE employees
SET salary = salary * 1.05
WHERE department_id = (
SELECT department_id
FROM employees
WHERE first_name = 'John' AND last_name = 'Doe'
);

10. 删除数据

基本DELETE语句

DELETE语句用于从表中删除记录。

-- 删除单条记录
DELETE FROM employees
WHERE employee_id = 10;

-- 删除多条记录
DELETE FROM employees
WHERE department_id = 5;

-- 删除所有记录（谨慎使用！）
DELETE FROM employees_backup;

使用子查询删除

可以在DELETE语句中使用子查询来过滤需要删除的记录。

-- 删除与John在同一部门的员工
DELETE FROM employees
WHERE department_id = (
SELECT department_id
FROM employees
WHERE first_name = 'John' AND last_name = 'Doe'
);

-- 删除工资低于平均工资的员工
DELETE FROM employees
WHERE salary < (
SELECT AVG(salary) FROM employees
);

TRUNCATE TABLE语句

TRUNCATE TABLE语句用于快速删除表中的所有记录，比DELETE语句更高效。

-- 清空表（谨慎使用！）
TRUNCATE TABLE employees_backup;

11. 创建表

基本CREATE TABLE语句

CREATE TABLE语句用于创建新的数据库表。

-- 创建employees表
CREATE TABLE employees (
employee_id INT PRIMARY KEY,
first_name VARCHAR(50),
last_name VARCHAR(50),
email VARCHAR(100),
salary DECIMAL(10, 2),
department_id INT
);

创建带有约束的表

可以在创建表时添加各种约束。

-- 创建带有约束的departments表
CREATE TABLE departments (
department_id INT PRIMARY KEY,
department_name VARCHAR(50) UNIQUE NOT NULL,
manager_id INT,
FOREIGN KEY (manager_id) REFERENCES employees(employee_id)
);

复制表结构

可以创建一个与现有表结构相同的新表。

-- 创建employees_backup表，复制employees表的结构
CREATE TABLE employees_backup AS
SELECT * FROM employees WHERE 1 = 0;

12. 约束

常见约束类型

约束用于确保数据的完整性和一致性。

  • PRIMARY KEY：主键约束，确保列值唯一且不为空
  • UNIQUE：唯一约束，确保列值唯一
  • NOT NULL：非空约束，确保列值不为空
  • FOREIGN KEY：外键约束，确保列值引用其他表的主键
  • CHECK：检查约束，确保列值满足特定条件
  • DEFAULT：默认约束，为列设置默认值

添加约束

可以在创建表时或之后添加约束。

-- 创建表时添加约束
CREATE TABLE products (
product_id INT PRIMARY KEY,
product_name VARCHAR(100) NOT NULL,
price DECIMAL(10, 2) CHECK (price >= 0),
category_id INT DEFAULT 1
);

-- 向现有表添加约束
ALTER TABLE products
ADD UNIQUE (product_name);

ALTER TABLE products
ADD FOREIGN KEY (category_id) REFERENCES categories(category_id);

删除约束

可以从表中删除约束。

-- 删除唯一约束
ALTER TABLE products
DROP CONSTRAINT UQ_products_product_name;

-- 删除外键约束
ALTER TABLE products
DROP FOREIGN KEY FK_products_categories;

13. 索引

创建索引

索引用于提高查询性能。

-- 创建单列索引
CREATE INDEX idx_employees_last_name
ON employees (last_name);

-- 创建复合索引
CREATE INDEX idx_employees_department_salary
ON employees (department_id, salary);

-- 创建唯一索引
CREATE UNIQUE INDEX idx_employees_email
ON employees (email);

删除索引

可以删除不再需要的索引。

-- 删除索引
DROP INDEX idx_employees_last_name ON employees;

14. 视图

创建视图

视图是基于SQL查询结果的虚拟表。

-- 创建简单视图
CREATE VIEW employee_view AS
SELECT first_name, last_name, email, salary
FROM employees
WHERE salary > 5000;

-- 创建带有别名的视图
CREATE VIEW department_summary AS
SELECT d.department_name AS 部门名称,
COUNT(e.employee_id) AS 员工数量,
AVG(e.salary) AS 平均工资
FROM employees e
INNER JOIN departments d ON e.department_id = d.department_id
GROUP BY d.department_name;

使用视图

可以像使用普通表一样使用视图。

-- 查询视图
SELECT * FROM employee_view;

SELECT 部门名称, 平均工资
FROM department_summary
ORDER BY 平均工资 DESC;

更新视图

可以更新满足特定条件的视图。

-- 更新视图
CREATE OR REPLACE VIEW employee_view AS
SELECT first_name, last_name, email, salary, department_id
FROM employees
WHERE salary > 6000;

-- 删除视图
DROP VIEW employee_view;

15. 存储过程

创建存储过程

存储过程是一组预编译的SQL语句，可以重复执行。

-- 创建简单存储过程
CREATE PROCEDURE GetAllEmployees()
AS
BEGIN
SELECT * FROM employees;
END;

-- 创建带有参数的存储过程
CREATE PROCEDURE GetEmployeesByDepartment(
@department_id INT
)
AS
BEGIN
SELECT first_name, last_name, email
FROM employees
WHERE department_id = @department_id;
END;

-- 创建带有输出参数的存储过程
CREATE PROCEDURE GetEmployeeCount(
@department_id INT,
@count INT OUTPUT
)
AS
BEGIN
SELECT @count = COUNT(*)
FROM employees
WHERE department_id = @department_id;
END;

执行存储过程

可以使用EXECUTE或EXEC命令执行存储过程。

-- 执行简单存储过程
EXECUTE GetAllEmployees;

-- 执行带有参数的存储过程
EXECUTE GetEmployeesByDepartment 10;

-- 执行带有输出参数的存储过程
DECLARE @employee_count INT;
EXECUTE GetEmployeeCount 10, @employee_count OUTPUT;
SELECT @employee_count AS 员工数量;

修改和删除存储过程

可以修改或删除存储过程。

-- 修改存储过程
ALTER PROCEDURE GetAllEmployees()
AS
BEGIN
SELECT first_name, last_name, email
FROM employees
ORDER BY last_name;
END;

-- 删除存储过程
DROP PROCEDURE GetAllEmployees;

16. 触发器

什么是触发器？

触发器是在特定事件发生时自动执行的SQL代码。

创建触发器

可以为INSERT、UPDATE或DELETE操作创建触发器。

-- 创建INSERT触发器
CREATE TRIGGER AfterEmployeeInsert
AFTER INSERT ON employees
FOR EACH ROW
BEGIN
INSERT INTO employee_audit (employee_id, action, change_time)
VALUES (NEW.employee_id, 'INSERT', NOW());
END;

-- 创建UPDATE触发器
CREATE TRIGGER AfterEmployeeUpdate
AFTER UPDATE ON employees
FOR EACH ROW
BEGIN
INSERT INTO employee_audit (employee_id, action, change_time)
VALUES (NEW.employee_id, 'UPDATE', NOW());
END;

-- 创建DELETE触发器
CREATE TRIGGER AfterEmployeeDelete
AFTER DELETE ON employees
FOR EACH ROW
BEGIN
INSERT INTO employee_audit (employee_id, action, change_time)
VALUES (OLD.employee_id, 'DELETE', NOW());
END;

删除触发器

可以删除触发器。

-- 删除触发器
DROP TRIGGER AfterEmployeeInsert;

17. 事务

事务的基本概念

事务是一组原子性的SQL操作，要么全部成功，要么全部失败。

事务的ACID特性

  • 原子性（Atomicity）：事务是一个不可分割的工作单位
  • 一致性（Consistency）：事务执行前后，数据库保持一致性状态
  • 隔离性（Isolation）：多个事务并发执行时，彼此隔离
  • 持久性（Durability）：事务一旦提交，其结果将永久保存

事务控制语句

可以使用COMMIT、ROLLBACK和SAVEPOINT语句控制事务。

-- 开始事务
START TRANSACTION;

-- 执行SQL操作
UPDATE accounts
SET balance = balance - 1000
WHERE account_id = 1;

UPDATE accounts
SET balance = balance + 1000
WHERE account_id = 2;

-- 提交事务
COMMIT;

-- 使用ROLLBACK
START TRANSACTION;
UPDATE accounts
SET balance = balance - 500
WHERE account_id = 1;
ROLLBACK; -- 撤销操作

-- 使用SAVEPOINT
START TRANSACTION;
INSERT INTO orders (order_date, customer_id)
VALUES (NOW(), 1);
SAVEPOINT order_inserted;
INSERT INTO order_items (order_id, product_id, quantity)
VALUES (LAST_INSERT_ID(), 10, 5);
ROLLBACK TO order_inserted; -- 回滚到保存点
COMMIT;

18. 权限管理

用户管理

可以创建和管理数据库用户。

-- 创建用户
CREATE USER 'new_user'@'localhost' IDENTIFIED BY 'password123';

-- 修改用户密码
ALTER USER 'new_user'@'localhost' IDENTIFIED BY 'new_password456';

-- 删除用户
DROP USER 'new_user'@'localhost';

权限管理

可以使用GRANT和REVOKE语句管理用户权限。

-- 授予权限
GRANT SELECT, INSERT, UPDATE
ON employees
TO 'new_user'@'localhost';

-- 授予所有权限
GRANT *.*
TO 'admin_user'@'localhost'
WITH GRANT OPTION;

-- 撤销权限
REVOKE INSERT, UPDATE
ON employees
FROM 'new_user'@'localhost';

-- 查看权限
SHOW GRANTS FOR 'new_user'@'localhost';

19. 高级查询技巧

窗口函数

窗口函数用于在查询结果集中执行计算，而不影响结果集的行数。

-- ROW_NUMBER()：为每行分配唯一序号
SELECT first_name, last_name, department_id, salary,
ROW_NUMBER() OVER (PARTITION BY department_id ORDER BY salary DESC) AS salary_rank
FROM employees;

-- RANK()：分配排名，相同值有相同排名
SELECT first_name, last_name, salary,
RANK() OVER (ORDER BY salary DESC) AS global_rank
FROM employees;

通用表表达式 (CTE)

CTE提供了一种创建临时结果集的方法，可以在主查询中引用。

-- 基本CTE
WITH high_salary_employees AS (
SELECT first_name, last_name, salary, department_id
FROM employees
WHERE salary > 8000
)
SELECT d.department_name, COUNT(*) AS high_salary_count
FROM high_salary_employees hse
INNER JOIN departments d ON hse.department_id = d.department_id
GROUP BY d.department_name;

PIVOT和UNPIVOT

PIVOT用于将行转换为列，UNPIVOT用于将列转换为行。

-- PIVOT示例
SELECT department_name, IT, HR, Finance
FROM (
SELECT d.department_name, e.first_name, d.department_name AS department
FROM employees e
INNER JOIN departments d ON e.department_id = d.department_id
) AS source_table
PIVOT (
COUNT(first_name)
FOR department IN (IT, HR, Finance)
) AS pivot_table;

EXISTS和NOT EXISTS

EXISTS和NOT EXISTS用于检查子查询是否返回结果。

-- 使用EXISTS
SELECT department_name
FROM departments d
WHERE EXISTS (
SELECT *
FROM employees e
WHERE e.department_id = d.department_id
AND e.salary > 10000
);

-- 使用NOT EXISTS
SELECT department_name
FROM departments d
WHERE NOT EXISTS (
SELECT *
FROM employees e
WHERE e.department_id = d.department_id
);
