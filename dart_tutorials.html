1. 变量声明与数据类型

Dart - 变量声明与初始化

特点：Dart支持var、final、const关键字声明变量，类型推导功能强大。注意事项：final变量只能赋值一次，const变量在编译时确定值。

# 变量声明
// 使用var声明变量（类型推导）
var name = 'Dart';
var age = 25;
var isActive = true;

// 显式声明类型
String message = 'Hello';
int count = 10;
double price = 19.99;

// final变量（只能赋值一次）
final String constantName = 'Fixed Value';

// const变量（编译时常量）
const int maxSize = 100;

2. 运算符与表达式

Dart - 算术运算符

Dart支持基本的算术运算符，包括加减乘除、取余、自增自减等。

# 算术运算符
var a = 10;
var b = 3;

print(a + b); // 13
print(a - b); // 7
print(a * b); // 30
print(a / b); // 3.333...
print(a ~/ b); // 3 (整数除法)
print(a % b); // 1 (取余)

// 自增自减
var c = 5;
print(c++); // 5 (后增)
print(++c); // 7 (前增)
print(c--); // 7 (后减)
print(--c); // 5 (前减)

Dart - 关系运算符与逻辑运算符

Dart提供了丰富的关系运算符和逻辑运算符，用于条件判断和逻辑组合。

# 关系运算符与逻辑运算符
// 关系运算符
var x = 5;
var y = 10;

print(x < y); // true
print(x > y); // false
print(x <= y); // true
print(x >= y); // false
print(x == y); // false
print(x != y); // true

// 逻辑运算符
var isTrue = true;
var isFalse = false;

print(isTrue && isFalse); // false (逻辑与)
print(isTrue || isFalse); // true (逻辑或)
print(!isTrue); // false (逻辑非)

3. 控制流语句

Dart - if-else语句

Dart的if-else语句与其他编程语言类似，支持条件判断和分支执行。

# if-else语句
var score = 85;

if (score >= 90) {
print('优秀');
} else if (score >= 80) {
print('良好');
} else if (score >= 60) {

print('不及格');

Dart - switch语句

Dart的switch语句支持字符串、数字等类型，每个case必须以break、return或throw结束。

# switch语句
var day = 'Monday';

switch (day) { case 'Monday': print('周一'); break; case 'Tuesday': print('周二'); break; case 'Wednesday': print('周三'); break; default: print('其他'); }

Dart - 循环语句

Dart支持for、while、do-while循环，以及for-in循环遍历集合。

# 循环语句
// for循环
for (var i = 0; i < 5; i++) {
print(i);
}

// while循环
var j = 0;
while (j < 5) {
print(j);
j++;
}

// do-while循环
var k = 0;
do {
print(k);
k++;
} while (k < 5);

// for-in循环
var list = ['a', 'b', 'c'];
for (var item in list) {
print(item);
}

4. 函数

Dart - 函数定义与调用

Dart支持函数定义、参数传递、返回值等功能，函数也是对象。

# 函数定义与调用
// 基本函数定义
int add(int a, int b) { return a + b; }
// 函数调用
var result = add(3, 5); print(result); // 8
// 箭头函数 (单行函数的简写)
int multiply(int a, int b) => a * b;
// 可选参数
void greet(String name, [String message]) { var msg = message ?? 'Hello'; print('$msg, $name!'); }
// 命名参数
void configure({required String host, int port = 8080}) { print('Host: $host, Port: $port'); }
// 调用命名参数函数
configure(host: 'localhost', port: 9090);

Dart - 匿名函数

Dart支持匿名函数（Lambda表达式），可以作为参数传递或赋值给变量。

# 匿名函数
// 定义匿名函数
var numbers = [1, 2, 3, 4, 5];
// 使用匿名函数作为forEach的参数
numbers.forEach((number) {
print(number);
});

// 使用箭头函数简写
numbers.forEach((number) => print(number));
// 赋值给变量
var square = (n) => n * n; print(square(5)); // 25

5. 类与对象

Dart - 类定义与对象创建

Dart是面向对象的语言，支持类定义、构造函数、继承等特性。

# 类定义与对象创建
// 类定义
class Person {
String? name;
int? age;

// 构造函数
Person(this.name, this.age);

// 方法
void introduce() {
print('Hi, I\'m $name, $age years old.');

Dart - 构造函数

Dart支持多种构造函数形式，包括命名构造函数、重定向构造函数等。

# 构造函数

Point.fromJson(Map<String, dynamic> json) : this(json['x'], json['y']);

// 使用不同构造函数
var p1 = Point(1.0, 2.0); var p2 = Point.origin(); var p3 = Point.fromJson({'x': 3.0, 'y': 4.0});

6. 接口与抽象类

Dart - 抽象类

Dart的抽象类使用abstract关键字定义，可以包含抽象方法和具体方法。

# 抽象类
abstract class Shape {
// 抽象方法
double calculateArea();

// 具体方法
void draw() {
print('Drawing shape');
}
}

class Circle extends Shape {
double radius;

Circle(this.radius);

@override
double calculateArea() {
return 3.14159 * radius * radius;
}
}

var circle = Circle(5.0); print(circle.calculateArea()); // 78.53975 circle.draw(); // Drawing shape

Dart - 接口

Dart中任何类都可以作为接口，使用implements关键字实现接口。

# 接口
class Drawable {
void draw() {
print('Drawing');
}
}

class Movable {
void move() {
print('Moving');
}
}

class Car implements Drawable, Movable {
@override
void draw() {
print('Drawing car');
}

@override
void move() {
print('Car moving');
}
}

var car = Car(); car.draw(); // Drawing car car.move(); // Car moving

7. 泛型

Dart - 泛型类与方法

Dart支持泛型，可以创建通用的类、方法和函数。

# 泛型类与方法
// 泛型类
class Box<T> {
T? value;

Box(this.value);

void setValue(T newValue) {
value = newValue;
}

T? getValue() {
return value;
}
}

// 使用泛型类
var stringBox = Box<String>('Hello'); var intBox = Box<int>(42);
print(stringBox.getValue()); // Hello print(intBox.getValue()); // 42
// 泛型方法
T findFirst<T>(List<T> items) { if (items.isEmpty) { return null as T; } return items[0]; }
var numbers = [1, 2, 3]; var firstNumber = findFirst(numbers); print(firstNumber); // 1

8. 集合类型

Dart - List集合

List是有序的对象集合，支持索引访问和修改。

# List集合
// 创建List
var fruits = ['apple', 'banana', 'orange']; var numbers = List<int>.filled(3, 0);
// 访问元素
print(fruits[0]); // apple

fruits[1] = 'grape';

// 检查元素是否存在
print(uniqueNumbers.contains(2)); // true

for (var number in uniqueNumbers) { print(number); }

9. 异步编程

Dart - Future与异步操作

Dart使用Future表示异步操作的结果，支持async/await语法。

# Future与异步操作
// 返回Future的异步函数
Future<String> fetchData() { return Future.delayed(Duration(seconds: 2), () { return 'Data fetched successfully'; }); }
// 使用then()处理Future结果
fetchData().then((data) { print(data); }).catchError((error) { print('Error: $error'); }).whenComplete(() { print('Operation completed'); });
// 使用async/await语法
void main() async { try { var data = await fetchData(); print(data); } catch (error) { print('Error: $error'); } finally { print('Operation completed'); } }

10. Stream编程

Dart - Stream与事件流

Stream表示一系列异步事件，支持监听和转换操作。

# Stream与事件流
// 创建Stream
Stream<int> countStream(int max) { return Stream.periodic(Duration(seconds: 1), (i) => i + 1).take(max); }
// 使用listen()监听Stream
countStream(5).listen( (data) => print(data), onError: (error) => print('Error: $error'), onDone: () => print('Stream completed') );
// 使用async/await与Stream
void main() async {
var stream = countStream(3);

// 使用await for循环
await for (var number in stream) {
print(number);
}

print('Stream processing finished');
}

11. 异常处理

Dart - 异常捕获与处理

Dart支持try-catch-finally异常处理机制，可以自定义异常类型。

# 异常捕获与处理
// 捕获异常
try { var result = 10 ~/ 0; // 除以零异常 print(result); } on IntegerDivisionByZeroException { print('Cannot divide by zero'); } on Exception catch (e) { print('Other exception: $e'); } finally { print('Finally block executed'); }
// 自定义异常
class InvalidInputException extends Exception {
final String message;

InvalidInputException(this.message);

@override
String toString() {
return 'InvalidInputException: $message';
}
}

void validateInput(String input) {
if (input.isEmpty) {
throw InvalidInputException('Input cannot be empty');
}
}

try {
validateInput('');
} catch (e) {
print(e);
}

14. 扩展方法

Dart - 扩展方法

扩展方法允许为现有类型添加新方法，而无需修改类型本身。

# 扩展方法
// 为String类型添加扩展方法
extension StringExtension on String { bool isPalindrome() { var reversed = this.split('').reversed.join(''); return this == reversed; }
String toTitleCase() { if (this.isEmpty) return '';
return this[0].toUpperCase() + this.substring(1).toLowerCase(); } }
// 使用扩展方法
var word = 'racecar'; print(word.isPalindrome()); // true
var sentence = 'hello world'; print(sentence.toTitleCase()); // Hello world
// 为int类型添加扩展方法
extension IntExtension on int { bool isEven() { return this % 2 == 0; }
bool isOdd() { return this % 2 != 0; } }
var number = 10; print(number.isEven()); // true print(number.isOdd()); // false

15. Mixins

Dart - Mixins

Mixins允许重用类的代码，而无需继承。

# Mixins
// 定义Mixin
mixin Flyable {
void fly() {
print('Flying');
}
}

mixin Swimmable {
void swim() {
print('Swimming');
}
}

// 使用Mixin
class Animal {
void eat() {
print('Eating');
}
}

class Duck extends Animal with Flyable, Swimmable {
void quack() {
print('Quacking');
}
}

// 使用带有Mixin的类
var duck = Duck(); duck.eat(); // Eating duck.fly(); // Flying duck.swim(); // Swimming duck.quack(); // Quacking
// 限制Mixin的使用
mixin Walkable on Animal {
void walk() {
print('Walking');
}
}

class Dog extends Animal with Walkable {
void bark() {
print('Barking');
}
}

16. 级联运算符

Dart - 级联运算符

级联运算符允许对同一个对象执行一系列操作。

# 级联运算符
// 定义一个类
class Person {
String? name;
int? age;
String? email;

void setName(String name) {
this.name = name;
}

void setAge(int age) {
this.age = age;
}

void setEmail(String email) {
this.email = email;
}

void printInfo() {
print('Name: $name, Age: $age, Email: $email');
}
}

// 不使用级联运算符
var person1 = Person(); person1.setName('Alice'); person1.setAge(25); person1.setEmail('alice@example.com'); person1.printInfo();
// 使用级联运算符
var person2 = Person() ..setName('Bob') ..setAge(30) ..setEmail('bob@example.com') ..printInfo();
// 级联运算符与构造函数
var person3 = Person() ..name = 'Charlie' ..age = 35 ..email = 'charlie@example.com' ..printInfo();
// 级联运算符与集合
var list = <int>[] ..add(1) ..add(2) ..addAll([3, 4, 5]) ..sort();
print(list); // [1, 2, 3, 4, 5]

17. 空安全

Dart - 空安全

空安全帮助防止空指针异常，是Dart 2.12+的重要特性。

# 空安全
// 非空类型（默认）
String name = 'Alice'; // 不能为null
// 可空类型
String? nullableName; // 可以为null nullableName = null; // 合法
// 空断言运算符
String nonNullableName = nullableName!; // 断言nullableName不为null
// 条件成员访问
int? length = nullableName?.length; // 如果nullableName为null，length也为null
// 空合并运算符
String displayName = nullableName ?? 'Anonymous'; // 如果nullableName为null，使用'Anonymous'
// 空合并赋值
nullableName ??= 'Default Name'; // 如果nullableName为null，赋值为'Default Name'
// 类型提升
if (nullableName != null) {
int len = nullableName.length; // Dart知道这里nullableName不为null
}
