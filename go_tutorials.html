1. 变量声明与初始化

Go - 变量声明、初始化和类型推导

特点：支持多种声明方式，包括标准声明、短变量声明、批量声明。注意事项：Go是静态类型语言，变量声明后类型不可更改，短变量声明 := 只能在函数内部使用。

# 变量声明
// 标准声明
var age int
var name string
var isActive bool

// 声明并初始化
var age = 25
var name = "张三"
var price float64 = 19.99

// 短变量声明（函数内部）
> age := 25
> name := "李四"
> isActive := true

// 批量声明
var (
    name = "王五"
    age = 30
    isActive = true
)

2. 基本数据类型

Go - 基本数据类型

特点：Go有丰富的内置类型，包括布尔型、数字型、字符串型。注意事项：不同类型的数值不能直接运算，需要显式转换，每个类型都有零值。

# 基本数据类型
// 布尔型
var b bool = true // 零值为 false

// 整型
var i8 int8 = 127 // 8位整数
var i16 int16 = 32767 // 16位整数
var i32 int32 = 2147483647 // 32位整数
var i64 int64 = 9223372036854775807 // 64位整数
var u uint = 42 // 无符号整数
var ui8 uint8 = 255 // 8位无符号整数
var r rune = 65 // 等价于 int32，用于表示 Unicode 码点
var bt byte = 255 // 等价于 uint8

// 浮点型
var f32 float32 = 3.14
var f64 float64 = 3.141592653589793

// 复数型
var c64 complex64 = complex(3, 4) // 3+4i
var c128 complex128 = complex(3.14, 2.71) // 3.14+2.71i

// 字符串型
var s string = "Hello, 世界"

// 零值示例
var uninitializedInt int // 零值为 0
var uninitializedBool bool // 零值为 false
var uninitializedString string // 零值为 ""

3. 控制流语句

Go - 控制流语句

特点：Go的控制流语句简洁明了，if语句条件不需要括号，switch语句不需要break，for语句功能强大。注意事项：Go没有while和do-while语句，都用for替代。

# 控制流语句
// if 语句
if age >= 18 {
    fmt.Println("成年人")
} else if age >= 13 {
    fmt.Println("青少年")
} else {
    fmt.Println("儿童")
}

// if 与初始化语句
if score := getScore(); score >= 90 {
    fmt.Println("优秀")
} else if score >= 80 {
    fmt.Println("良好")
}

// for 循环
for i := 0; i < 10; i++ {
    fmt.Println(i)
}

// while 等价写法
for count > 0 {
    fmt.Println(count)
    count--
}

// 无限循环
for {
    fmt.Println("无限循环")
    break // 避免真正无限循环
}

// switch 语句
switch day {
case "Monday", "Tuesday", "Wednesday", "Thursday", "Friday":
    fmt.Println("工作日")
case "Saturday", "Sunday":
    fmt.Println("周末")
default:
    fmt.Println("未知")
}

// 类型 switch
switch v := value.(type) {
case int:
    fmt.Println("整数类型", v)
case string:
    fmt.Println("字符串类型", v)
default:
    fmt.Println("其他类型")
}

4. 函数定义与使用

Go - 函数定义与使用

特点：支持多返回值、命名返回值、变参函数、匿名函数和闭包。注意事项：Go函数是一等公民，可以作为值传递，函数名首字母大小写决定可见性。

# 函数定义与使用
// 基本函数定义
func add(a int, b int) int {
    return a + b
}

// 多返回值
func divide(a, b float64) (float64, error) {
    if b == 0 {
        return 0, errors.New("除数不能为零")
    }
    return a / b, nil
}

// 命名返回值
func calculate(a, b int) (sum, product int) {
    sum = a + b
    product = a * b
    return // 返回 sum 和 product
}

// 变参函数
func sum(numbers ...int) int {
    total := 0
    for _, num := range numbers {
        total += num
    }
    return total
}

// 匿名函数
var multiply = func(a, b int) int {
    return a * b
}

// 闭包
func counter() func() int {
    i := 0
    return func() int {
        i++
        return i
    }
}

// 函数作为参数
func apply(fn func(int) int, value int) int {
    return fn(value)
}

5. 数组和切片

Go - 数组和切片

特点：数组是值类型，长度固定；切片是引用类型，动态大小。注意事项：切片是Go中更常用的序列类型，提供了灵活的数据操作方式。

# 数组和切片
// 数组定义
var arr1 [5]int // 长度为5的整数数组，元素为零值
var arr2 = [3]string{"a", "b", "c"} // 初始化数组
var arr3 = [...]int{1, 2, 3, 4} // 自动推断长度

// 切片定义
var slice1 []int // 零值为 nil
var slice2 = []int{1, 2, 3} // 字面量创建切片
var slice3 = arr1[1:4] // 从数组切片
slice4 := make([]int, 5) // 使用 make 创建切片
slice5 := make([]int, 3, 10) // 长度3，容量10

// 切片操作
slice1 = append(slice1, 10) // 添加元素
slice1 = append(slice1, 20, 30) // 添加多个元素
slice1 = append(slice1, slice2...) // 追加另一个切片

// 切片的切片
subSlice := slice1[1:3] // [start:end]
subSlice2 := slice1[1:4:6] // [start:end:capacity]

// 遍历
for i, value := range slice1 {
    fmt.Printf("索引: %d, 值: %d\n", i, value)
}

// 切片的长度和容量
fmt.Printf("长度: %d, 容量: %d\n", len(slice1), cap(slice1))

6. 映射(Map)

Go - 映射(Map)

特点：键值对集合，动态大小，引用类型。注意事项：map是引用类型，需要初始化后才能使用，访问不存在的键会返回零值，删除不存在的键不会报错。

# 映射(Map)
// Map定义
var m1 map[string]int // 零值为 nil
m2 := make(map[string]int) // 使用 make 创建
m3 := map[string]int{ // 字面量创建
    "apple": 5,
    "banana": 3,
    "orange": 8,
}

// 基本操作
m2["key1"] = 100 // 插入/更新
value := m2["key1"] // 访问

// 检查键是否存在
if value, exists := m2["key1"]; exists {
    fmt.Printf("键存在，值为: %d\n", value)
} else {
    fmt.Println("键不存在")
}

// 删除键值对
delete(m2, "key1")

// 遍历Map
for key, value := range m3 {
    fmt.Printf("键: %s, 值: %d\n", key, value)
}

// 获取Map长度
fmt.Printf("Map长度: %d\n", len(m3))

// 嵌套Map
nestedMap := map[string]map[string]int{
    "fruits": {
        "apple": 5,
        "orange": 3,
    },
    "vegetables": {
        "carrot": 10,
        "potato": 7,
    },
}

7. 结构体和方法

Go - 结构体和方法

特点：结构体是值类型，支持嵌入字段（类似继承），方法可以定义在任何类型上。注意事项：方法接收者分为值接收者和指针接收者，选择合适的接收者类型很重要。

# 结构体和方法
// 结构体定义
type Person struct {
    Name string
    Age int
    Email string
}

// 创建结构体实例
p1 := Person{"张三", 25, "zhang@example.com"} // 按顺序初始化
p2 := Person{Name: "李四", Age: 30} // 字段名初始化
p3 := Person{} // 零值初始化
var p4 Person // 变量声明

// 结构体指针
p5 := &Person{Name: "王五", Age: 35}

// 方法定义（值接收者）
func (p Person) Greet() string {
    return fmt.Sprintf("你好，我是%s，今年%d岁", p.Name, p.Age)
}

// 方法定义（指针接收者）
func (p *Person) SetAge(age int) {
    p.Age = age
}

// 嵌入字段（类似继承）
type Address struct {
    Street string
    City string
}

type Employee struct {
    Person // 嵌入Person结构体
    Position string
    Address // 嵌入Address结构体
}

// 接口实现
type Speaker interface {
    Speak() string
}

func (p Person) Speak() string {
    return "Hello, my name is " + p.Name
}

8. 接口

Go - 接口

特点：Go接口是隐式实现的，不需要显式声明实现关系，支持接口嵌套。注意事项：接口是Go实现多态的主要方式，空接口interface{}可以表示任意类型。

# 接口
// 基本接口定义
type Writer interface {
    Write([]byte) (int, error)
}

type Reader interface {
    Read([]byte) (int, error)
}

// 接口嵌套
type ReadWriter interface {
    Reader
    Writer
}

// 实现接口的结构体
type FileWriter struct {
    filename string
}

func (fw FileWriter) Write(data []byte) (int, error) {
    return len(data), nil // 简化的写入实现
}

// 空接口
var anything interface{} = 42 // 可以存储任何类型
var anything2 interface{} = "hello"
var anything3 interface{} = []int{1, 2, 3}

// 类型断言
if str, ok := anything2.([string     fmt.Printf("字符串值: %s\n", str)
}

// 类型选择
func do(v interface{}) {
    switch t := v.(type) {
    case string:
        fmt.Printf("字符串: %s\n", t)
    case int:
        fmt.Printf("整数: %d\n", t)
    default:
        fmt.Println("未知类型")
    }
}

9. 并发编程(Goroutines和Channels)

Go - 并发编程

特点：Goroutines轻量级线程，Channels用于Goroutines间通信，支持select语句处理多个Channel。注意事项：并发是Go的核心特性，正确使用Channel可以避免竞态条件。

# 并发编程
// Goroutine 基本使用
func say(s string) {
    for i := 0; i < 3; i++ {
        fmt.Println(s)
        time.Sleep(time.Millisecond * 100)
    }
}

// 启动Goroutine
go say("世界")
say("你好")

// Channel 基本操作
ch := make(chan string)

go func() {
    ch <- "Hello from goroutine" // 发送数据到Channel
}()

msg := <-ch // 从Channel接收数据

// 带缓冲的Channel
bufferedCh := make(chan int, 2)
bufferedCh <- 1
bufferedCh <- 2
fmt.Println(<-bufferedCh) // 输出: 1

// 遍历Channel
func fibonacci(n int, c chan int) {
    x, y := 0, 1
    for i := 0; i < n; i++ {
        c <- x
        x, y = y, x+y
    }
    close(c) // 关闭Channel
}

func main() {
    c := make(chan int, 10)
    go fibonacci(cap(c), c)
    for i := range c { // 遍历直到Channel关闭
        fmt.Println(i)
    }
}

// Select 语句
func fibonacciWithSelect(c, quit chan int) {
    x, y := 0, 1
    for {
        select {
        case c <- x:
            x, y = y, x+y
        case <-quit:
            fmt.Println("退出")
            return
        }
    }
}

10. 错误处理

Go - 错误处理

特点：Go不使用异常机制，而是通过多返回值返回错误，提供了errors包创建错误。注意事项：错误处理是Go的重要特性，应该始终检查函数返回的错误值。

# 错误处理
// 错误处理基本模式
func divide(a, b float64) (float64, error) {
    if b == 0 {
        return 0, errors.New("除数不能为零")
    }
    return a / b, nil
}

// 使用错误
result, err := divide(10, 0)
if err != nil {
    fmt.Println("错误:", err)
    return
}
fmt.Printf("结果: %.2f\n", result)

// 自定义错误类型
type CustomError struct {
    Code int
    Msg string
}

func (e *CustomError) Error() string {
    return fmt.Sprintf("错误代码: %d, 消息: %s", e.Code, e.Msg)
}

// 使用fmt.Errorf格式化错误
func validateAge(age int) error {
    if age < 0 {
        return fmt.Errorf("年龄不能为负数: %d", age)
    }
    if age > 150 {
        return fmt.Errorf("年龄不能超过150岁: %d", age)
    }
    return nil
}

// 错误检查辅助函数
func checkError(err error) {
    if err != nil {
        panic(err)
    }
}

12. Go 1.25 语言规范变化

Go 1.25 - 移除"核心类型"概念

特点：Go 1.25从语言规范中移除了"核心类型"（Core Types）这一概念，这是Go 1.18引入泛型时的临时解决方案。注意事项：这一变化不影响现有代码，但使语言规范更清晰易懂。

# 核心类型概念的移除
// Go 1.18-1.24: 泛型中的核心类型概念（现在移除）
func min[T comparable](a, b T) T {
    if a < b {
        return a
    }
    return b
}

// 这类代码在Go 1.25中仍然有效，但规范描述更清晰
func max[T comparable](a, b T) T {
    return func() T {
        if a > b {
            return a
        }
        return b
    }()
}

// Go 1.25中泛型的使用没有变化，只是规范更清晰
type Number interface {
    int | int32 | int64 | float32 | float64
}

func add[T Number](a, b T) T {
    return a + b
}

16. Go 1.25 新的并发测试包

Go 1.25 - testing/synctest 包

特点：Go 1.25引入testing/synctest包，用于测试并发代码。注意事项：这个包提供在独立"气泡"中运行测试的能力，以及伪造时钟功能。

# 新的并发测试包
// test_example_test.go
package main

import (
    "context"
    "testing"
    "time"
)

// 模拟一个需要并发测试的服务
type Counter struct {
    count int
}

func (c *Counter) Increment() {
    c.count++
}

func (c *Counter) Get() int {
    return c.count
}

func TestConcurrentCounter(t *testing.T) {
    counter := &Counter{}

    // 启动多个goroutine同时增加计数器
    for i := 0; i < 10; i++ {
        go func() {
            counter.Increment()
        }()
    }

    // 等待所有goroutine完成
    time.Sleep(time.Millisecond * 100)

    if counter.Get() != 10 {
        t.Errorf("Expected count to be 10, got %d", counter.Get())
    }
}

// Go 1.25中使用testing/synctest（概念示例）
import "golang.org/x/exp/synctest" // 假设的导入路径

func TestWithSynctest(t *testing.T) {
    // 在独立的"气泡"中运行测试
    b := synctest.New(t)

    b.Run("parallel test", func(tb *testing.T) {
        tb.Parallel() // 在气泡中并行运行
        // 测试代码
    })
}

17. 指针

Go - 指针

特点：指针是存储变量内存地址的变量。Go支持指针但没有指针运算，更安全。注意事项：使用&获取变量地址，*解引用指针，指针可避免大对象拷贝，提高性能。

# 指针的使用
// 基本指针操作
var num = 42
var ptr *int = &num // 获取num的地址

fmt.Printf("num的值: %d\n", num) // 42
fmt.Printf("num的地址: %p\n", &num) // 类似 0x1040a130
fmt.Printf("ptr的值(地址): %p\n", ptr) // 与&num相同
fmt.Printf("ptr指向的值: %d\n", *ptr) // 42

// 通过指针修改值
*ptr = 100

func selectExample() {

    ch2 := make(chan string)

        }
    }
}

import (
    "fmt"

    Name string `json:"name"`
    Age int `json:"age"`
    Email string
}

func reflectStruct() {
    person := Person{"Alice", 30, "alice@example.com"}
    rv := reflect.ValueOf(person)
    rt := reflect.TypeOf(person)

    for i := 0; i < rv.NumField(); i++ {
        field := rt.Field(i)
        value := rv.Field(i)

        fmt.Printf("字段: %s, 类型: %v, 值: %v",
            field.Name, field.Type, value)

        // 获取tag

    }

func withCancelExample() {
    ctx, cancel := context.WithCancel(context.Background())
    defer cancel()

    go func(ctx context.Context) {
        for {
            select {
            case <-ctx.Done():

        }

                fmt.Println("工作中...")

            if result != tt.expected {
                t.Errorf("Add(%d, %d) = %d; want %d",
                    tt.a, tt.b, result, tt.expected)
            }
        })
    }
}

28. 错误处理

Go - 错误处理

特点：Go通过返回值显式处理错误，不使用异常机制。注意事项：始终检查错误返回值，使用errors包创建自定义错误，合理处理和传播错误。

# 基本错误处理
package main

import (
    "errors"
    "fmt"
)

func divide(a, b float64) (float64, error) {
    if b == 0 {
        return 0, errors.New("除数不能为零")
    }
    return a / b, nil
}

func main() {
    result, err := divide(10, 2)
    if err != nil {
        fmt.Println("错误: ", err)
        return
    }
    fmt.Printf("结果: %.2f\n", result)
}
# 自定义错误类型
type CustomError struct {
    Code int
    Msg string
}

func (e *CustomError) Error() string {
    return fmt.Sprintf("错误代码: %d, 错误信息: %s", e.Code, e.Msg)
}

func validateAge(age int) error {
    if age < 0 || age > 150 {
        return &CustomError{Code: 400, Msg: "年龄不在有效范围内"}
    }
    return nil
}

func main() {
    err := validateAge(-5)
    if err != nil {
        fmt.Println(err) // 输出自定义错误信息
        if customErr, ok := err.(*CustomError); ok {
            fmt.Printf("捕获到自定义错误，代码: %d\n", customErr.Code)
        }
    }
}
# 错误包装
import (
    "fmt"
    "io"
)

// Go 1.13+ 错误包装
func readFile(filename string) error {
    data, err := os.ReadFile(filename)
    if err != nil {
        return fmt.Errorf("读取文件失败: %w", err)
    }
    fmt.Println("文件内容: ", string(data))
    return nil
}

func processFile(filename string) error {
    err := readFile(filename)
    if err != nil {
        return fmt.Errorf("处理文件时发生错误: %w", err)
    }
    return nil
}

func main() {
    err := processFile("nonexistent.txt")
    if err != nil {
        fmt.Println("错误: ", err)
        // 检查原始错误
        if errors.Is(err, os.ErrNotExist) {
            fmt.Println("文件不存在")
        }
    }
}
